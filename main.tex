    \subsection*{1. Определение числового ряда. Сходимость}
    \addcontentsline{toc}{subsection}{1. Определение числового ряда. Сходимость}
    \par{\textbf{Def 1: }\text{Числовым рядом называется пара последовательностей $\{a_n\}, \{S_N\}$, где} \\\text{$S_N=\displaystyle\sum_{n=1}^N a_{n}$. $S_N$ - частичные суммы, $a_n$ - члены ряда.}
    \\ \text{$\displaystyle\sum_{n=1}^\infty a_{n}$ - обозначение, !НО не определение.}}
    \vspace{1cm}
    \par{\textbf{Def 2}: \text{Ряд называется сходящимся, если $\exists$ конечный $\displaystyle\lim_{N\to\infty} S_N$ = S,}
    \\ \text{S - называется суммой ряда.}}
    \vspace{1cm}
    \par{\textbf{Def 3}: \text{Если $\nexists$ конечного $\displaystyle\lim_{N\to\infty} S_N$, то ряд называется расходящимся.}}
    \vspace{0.5cm}
    \par{\textbf{Примеры:} \\
        \begin{equation*}
        \sum_{n=1}^{\infty} q^{n} =  \lim_{N\to\infty} (q+q^2+...+q^N) =\lim_{N\to\infty} q \frac{1-q^N}{1-q} = \begin{cases} \frac{1}{1-q}, |q| < 1 \\ \text{расходится, |q| $\geq$ 1} \end{cases}
        \end{equation*}
    \begin{equation*}
            \sum_{n=1}^{\infty}{1\over{n!}} = e-1
        \end{equation*}
        \begin{equation*}
            \sum_{n=1}^{N} \ln{n} = N\ln{N}-N+\gamma + o(1) - \text{расходится}
        \end{equation*}
        }
        \par{Дзета-функция Римана:
        \begin{equation*}
            \sum_{n=1}^{\infty} {{1}\over{n^p}} = \zeta(p) - \text{Дзета-функция Римана}
        \end{equation*}}
        \begin{itemize}
            \item \par{\textbf{p<0}
            \begin{equation*}
                S_N = 1 + {{1}\over{2^p}}+...+{{1}\over{N^p}}=1+2^{-p>0}+...+N^{-p>0} > N \xrightarrow{N\to\infty} \infty
            \end{equation*}}
            \item \par{\textbf{$p \in [0, 1)$}
            \begin{equation*}
                S_N = 1 + {{1}\over{2^p}}+...+{{1}\over{N^p}}>{{1}\over{N^p}}+...+{{1}\over{N^p}}={{N}\over{N^p}}=N^{1-p} \xrightarrow{N\to\infty} \infty
            \end{equation*}}
            \newpage
            \item \par{\textbf{p=1}
            \begin{equation*}
                S_N = 1 + {{1}\over{2}}+...+{{1}\over{N}}=1+{{1}\over{2}}+({{1}\over{3}}+{{1}\over{4}})+({{1}\over{5}}+{{1}\over{6}}+{{1}\over{7}}+{{1}\over{8}})+...+(...+{{1}\over{2^n}})+\text{остат} > 
                \end{equation*}
                \begin{equation*}
                > 1 + {{1}\over{2}}+{{1}\over{4}}+{{1}\over{4}}+{{1}\over{8}}+{{1}\over{8}}+{{1}\over{8}}+{{1}\over{8}}+...+{{1}\over{(2^m)}}2^{m-1}=1+{{M}\over{2}}\xrightarrow{N\to\infty} \infty
            \end{equation*}}
            \item \par{\textbf{p>1}
            \begin{equation*}
                \text{Рассмотрим } S_{2^{N-1}}=1+({{1}\over{2^p}}+{{1}\over{3^p}})+...+({{1}\over{(2^{N-1}})^p}+{{1}\over{(2^{N-1}}+1)^p}+...+{{1}\over{(2^{N}}+1)^p})<
            \end{equation*}
            \textit{(выбираем и группируем так, чтобы каждое слагаемое начиналось со степени двойки)}
            \begin{equation*}
            < 1 + {{1}\over{2^p}}2+{{1}\over{4^p}}4 + \ldots+{{1}\over{(2^{N-1})^p}}2^{N-1}=1+2^{1-p}+...+(2^{N-1})^{1-p}=
            \end{equation*}
            \begin{equation*}
            =(2^{1-p})^0+(2^{1-p})^1+(2^{1-p})^2+...+(2^{1-p})^{N-1}<{{1}\over{1-{{1}\over2^{p-1}}}}=z
            \end{equation*}
            \textit{ (применили формулу геометрической прогрессии)}\\
            \text{Пусть $M\in\mathbb{N}$: $2^{N-1}\leq M < 2^N, S_M \leq S_{2^{N-1}} < z$, т.о. вся последовательность}\\
            \text{ограничена числом z $\Rightarrow \exists \lim$.}}
        \end{itemize}
    \subsection*{2. Простейшие свойства}
    \addcontentsline{toc}{subsection}{2. Простейшие свойства}
    \par{\textbf{Лемма 1.} \text{Добавление или отбрасывание конечного числа элементов ряда не ме-}\\
    \text{няет его сходимость.} \\ 
    \textbf{Доказательство:} \text{Рассмотрим $\displaystyle\sum_{n=1}^{\infty} a_{n}\ , S_n$ и  $\displaystyle\sum_{n=1}^{\infty} \tilde{a}_{n}$, $\tilde{S}_n$ и пусть отбросили M слагаемых}\\ 
    \text{и $N_0$ - максимальный номер отброшенного слагаемого.} \\
    \text{Пусть N > $N_0$:}}
    \par{
    \begin{equation*}
    S_N = S_{N_0} + \sum_{n=N_0+1}^{N} a_{n} \text{ , }
    \end{equation*}
    \begin{equation*}
    \tilde{S}_{N-M} = \tilde{S}_{{N_0}-M} + \sum_{n=N_0+1}^{N}a_{n} \text{ - выкинули M слагаемых.}
    \end{equation*}
    \text{Выразим сумму и приравняем, а затем перенесём через знак равенства:}
    \begin{equation*}
    S_N=\tilde{S}_{N-M}-\tilde{S}_{N_0-M}+S_{N_0} 
    \end{equation*}
    \text{Заметим, что последние два слагаемые не зависят от выбора N, поэтому} \\
    \text{являются const, а первое слагаемое является лишь сдвигом на M слагаемых.} \\
    \text{Тогда в пределе $\lim{S_N}$ и $\lim{\tilde{S}_{N-M}}$ $\exists$ или $\nexists$ одновременно.} \\
    \text{Для добавления доказательство аналогичное, только значки $\sim$ меняются местами.}
    }
    \par{{\raggedleft ч.т.д.\par}}
    \par{{\textbf{Лемма 2.} Если ряд $\displaystyle\sum_{n=1}^{\infty} a_{n}$  сходится, то и ряд  $\displaystyle\sum_{n=1}^{\infty} ca_{n}\ $(c $\in$ $\mathbb{R}$) сходится, $\\$ причём $\displaystyle\sum_{n=1}^{\infty} ca_{n}$ = c$\displaystyle\sum_{n=1}^{\infty} a_{n}. \ $ }\\
    }
    \newpage
    \par{{\textbf{Лемма 3.} Если ряды $\displaystyle\sum_{n=1}^{\infty} a_{n}$ и $\displaystyle\sum_{n=1}^{\infty} b_{n}$ сходятся, то и ряды  $\displaystyle\sum_{n=1}^{\infty}(a_{n} \pm b_{n})\ $ сходятся, $\\$ причём $\displaystyle\sum_{n=1}^{\infty}(a_{n} \pm b_{n})\ $ = $\displaystyle\sum_{n=1}^{\infty} a_{n} \pm \displaystyle\sum_{n=1}^{\infty} b_{n}$.\\
    }
    \textit{(доказательство лемм 2 и 3 следует из свойств пределов, необходимо рассмотреть конечные суммы)}}
    \vspace{0.5cm}
    \par{\textbf{Лемма 4.} Если ряд сходится, то общий член ряда стремится к нулю. \\
    \textbf{Доказательство: } \text{Рассмотрим $a_n$: $a_n = S_n - S_{n-1}$, т.к. ряд сходится, то его частич-} \\ \text{ные суммы стремятся к S $\Rightarrow S_n - S_{n-1} \rightarrow S-S=0$.}
    }
    \par{{\raggedleft ч.т.д.\par}}
    \text{ОБРАТНОЕ НЕВЕРНО! Пример: $\displaystyle\sum_{n=1}^{\infty} {{1}\over{n}}$ - расходится, хотя ${{1}\over{n}} \xrightarrow{N\to\infty} 0$. }
    \vspace{0.5cm}

    \addcontentsline{toc}{subsubsection}{Критерий Коши для ряда}
    \par{\textbf{Критерий Коши для ряда} \\
    \text{$\displaystyle\exists \lim_{N\to\infty} S_N \Leftrightarrow \forall\epsilon > 0 \ $ $\displaystyle\exists N_0 \in \mathbb{N}\ $: $\displaystyle\forall m > N_0, k \in \mathbb{N} \ $ верно $|S_{m+k}-S_m|=|a_{m+1}+...+a_{m+k}| <\epsilon$} \\
    \text{Доказательство следует из критерия Коши для последовательностей.}\\}
    \vspace{0.3cm}
    \par{
    \textbf{Пример:} \text{$\displaystyle\sum_{n=1}^{\infty}\frac{\sin{nx}}{2^n}, x\in \mathbb{R}$.} \\
    \text{Фикс. $\epsilon > 0$ и рассмотрим $\displaystyle|S_{m+k}-S_m| = |\frac{\sin{(n+1)x}}{2^{n+1}}+...+\frac{\sin{(n+k)x}}{2^{n+k}}| \leq \ $} \\\text{$\displaystyle\leq|\frac{\sin{(n+1)x}}{2^{n+1}}|+...+|\frac{\sin{(n+k)x}}{2^{n+k}}|\leq(|sin|\leq1)\leq \frac{1}{2^{n+1}}+...+\frac{1}{2^{n+k}} < \ $ (геом.прогрессия) <} \\
    \text{$\displaystyle<\frac{1}{2^{n+1}}\frac{1}{1-{{1}\over{2}}}=\frac{1}{2^n}<\epsilon\Rightarrow-n\ln{2}<\ln{\epsilon}\Rightarrow n>\frac{\ln{1\over{\epsilon}}}{\ln{2}}\Rightarrow N_0=[\frac{\ln{1\over{\epsilon}}}{\ln{2}}]+1. \ $} \\
    }

    \section*{$\S$2. Ряды с положительными членами}
    \addcontentsline{toc}{section}{$\S$2. Ряды с положительными членами}
    \subsection*{1. Определение. Признаки сравнения}
    \addcontentsline{toc}{subsection}{1. Определение. Признаки сравнения}
    \par{\textbf{Def 1:} \text{Если $a_n\geq0$, то ряд $\displaystyle\sum_{n=1}^{\infty}a_n \ $} называется рядом с положительными членами.}
    \vspace{0.5cm}
    \addcontentsline{toc}{subsubsection}{Лемма. Критерий сходимости положительного ряда}
    \par{\textbf{Лемма. Критерий сходимости положительного ряда} \\
    \text{Пусть $a_n\geq0$, тогда $\displaystyle\sum_{n=1}^{\infty} a_{n} \ $ сходится $\Leftrightarrow S_N$ ограничено.}\\
    \textbf{Доказательство: } \text{т.к. $S_N-S_{N-1}=a_n\geq0$, то $S_N$ - монотонно возрастает. По теореме} \\\text{о "пределе монотонной последовательности" \  $\exists\lim{S_N}\Leftrightarrow S_N$ ограничена.}
    }
    \vspace{0.7cm}
    \addcontentsline{toc}{subsubsection}{Теорема. Признак сравнения}
    \par{\textbf{Теорема. "Признак сравнения"} \\
    \text{Пусть, начиная с некоторого номера, $0\leq a_n\leq b_n$, тогда если ряд $\displaystyle\sum_{n=1}^{\infty}b_n \ $ сходится,} \\\text{тогда и ряд $\displaystyle\sum_{n=1}^{\infty}a_n\ $ сходится. А если ряд $\displaystyle\sum_{n=1}^{\infty}a_n\ $ расходится, то и ряд $\displaystyle\sum_{n=1}^{\infty}b_n\ $ расходится.} \\
        \textbf{Доказательство: } \text{согласно лемме 1 $\S$1 можно отбросить члены ряда так, чтобы} \newpage \text{выполнялось следующее: $0\leq a_n\leq b_n$ $\forall n\in\mathbb{N}$. Пусть $\displaystyle S_N=\sum_{n=1}^{\infty}a_n, \tilde{S}_N=\sum_{n=1}^{\infty}b_n $,}\\\text{тогда $S_N\leq \tilde{S}_N$ $\forall N\in \mathbb{N}$.} \\}
        \vspace{0cm}
        \begin{itemize}
            \item \par{
        \text{$\displaystyle\sum_{n=1}^{\infty}b_n\ $ сходится $\Leftrightarrow\displaystyle\exists\lim_{N\rightarrow\infty}{\tilde{S}_N}$. По лемме Критерий сходимости положительного ряда} \\
        \text{$\tilde{S}_N$ ограничена, тогда $S_N\leq\tilde{S}_N\leq c$, тогда $S_N$ ограничена. По той же лемме} \\\text{$S_N$ ограничена $\displaystyle\Leftrightarrow \sum_{n=1}^{\infty}a_n\ $ сходится}.}
        \item \par{
        \text{Пусть $\displaystyle\sum_{n=1}^{\infty}a_n\ $ расходится. По лемме Критерий сходимости положительного ряда}\\
        \text{это происходит тогда и только тогда, когда $S_N$ неограничена сверху. Т.е.}\\
        \text{$\forall K\in\mathbb{R}$ $\exists N_0\in\mathbb{N}$: $N>N_0 \Rightarrow S_N>K \Rightarrow \tilde{S}_N\geq S_N>K$ $\forall N>N_0 \Rightarrow$} \\\text{$\Rightarrow \tilde{S}_N$ неограничена сверху $\Leftrightarrow\displaystyle\sum_{n=1}^{\infty}b_n\ $ расходится по Критерию сходимости.}
        }
        \end{itemize}
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{1cm}
        \par{
        \textbf{Следствие.} \text{Пусть $a_n>0$, $b_n>0$ и $\displaystyle\lim_{n\rightarrow\infty}{\frac{a_n}{b_n}}=K$, где $0\leq K\leq +\infty$.}}
        \begin{enumerate}
            \item \par{\text{Если K<+$\infty$ (конечно), то из $\displaystyle\sum_{n=1}^{\infty}b_n$ сходится $\displaystyle\Rightarrow\sum_{n=1}^{\infty}a_n\ $ сходится.}}
            \item \par{\text{Если K>0, то из $\displaystyle\sum_{n=1}^{\infty}b_n$ расходится $\displaystyle\Rightarrow\sum_{n=1}^{\infty}a_n\ $ расходится.}}
            \item \par{\text{Если 0<K<+$\infty$ (конечно), то $\displaystyle\sum_{n=1}^{\infty}a_n$ и $\displaystyle\sum_{n=1}^{\infty}b_n\ $ сходятся/расходятся одновременно.}}
        \end{enumerate}
        \par{\textbf{Доказательтство:}}
        \begin{enumerate}
            \item \par{\text{Пусть K конечно, т.е. $\forall\epsilon>0$ $\exists N\in\mathbb{N}:$ $\forall n>N$ $K-\epsilon<\frac{a_n}{b_n}<K+\epsilon$ $\Rightarrow a_n<(K+\epsilon)b_n$.}\\
            \text{По лемме 2 из $\S$1 если $\displaystyle\sum_{n=1}^{\infty}b_n\ $ сходится, то и $\displaystyle\sum_{n=1}^{\infty}(K+\epsilon)b_n\ $, т.к. $K+\epsilon = const$,}\\
            \text{тогда и ряд $\displaystyle\sum_{n=1}^{\infty}a_n\ $ сходится (по теореме Признак сравнения).}\\\text{Действительно, $0<a_n<(K+\epsilon)b_n$ - все условия для применения этой теоремы.}
            }
            \item \par{\text{Если К>0, то $\lim{\frac{b_n}{a_n}}$ конечен. Пусть $\displaystyle\sum_{n=1}^{\infty}b_n\ $ расходится, а ряд $\displaystyle\sum_{n=1}^{\infty}a_n\ $ сходится.}\\\text{Тогда немедленно наступает $\lightning$ с п.1, ведь если знаменатель сходится, то и числитель}\\\text{сходится, а у нас $\displaystyle\sum_{n=1}^{\infty}a_n\ $(в знаменателе) сходится, а $\displaystyle\sum_{n=1}^{\infty}b_n\ $(в числителе) расходится.}
            }
            \item \par{\text{доказательство следует из п.1 и п.2 этого следствия.}}
        \end{enumerate}
        \par{{\raggedleft ч.т.д.\par}}
        \newpage
        \par{\textbf{Пример: } \text{$x_n=2\sqrt{n}-1-{{1}\over{\sqrt{2}}}-...-{{1}\over{\sqrt{n}}}$. $\displaystyle\sum_{n=1}^{\infty}(x_{n+1}-x_n)$.}\\
        \begin{equation*}
            S_N=(x_2-x_1)+(x_3-x_2)+...+(x_{N+1}-x_N)=x_{N+1}-x_1
        \end{equation*}
        \begin{equation*}
            x_{n+1}-x_n=2\sqrt{n+1}-2\sqrt{n}-{{1}\over{\sqrt{n+1}}}=\textit{(разность квадратов)}=2\frac{(n+1)-(n)}{\sqrt{n+1}+\sqrt{n}}-{{1}\over{\sqrt{n+1}}}=
        \end{equation*}
        \begin{equation*}
            ={{2}\over{\sqrt{n+1}}+\sqrt{n}}-{{1}\over{\sqrt{n+1}}}=\frac{\sqrt{n+1}-\sqrt{n}}{\sqrt{n+1}(\sqrt{n+1}+\sqrt{n})}=\textit{(разность квадратов)}=
        \end{equation*}
        \begin{equation*}
            ={{1}\over{\sqrt{n+1}(\sqrt{n+1}+\sqrt{n})^2}} \sim {{1}\over{4n^{1.5}}}\text{ т.е. предел их отношения равен 1.}
        \end{equation*}
        \text{$\displaystyle\sum_{n=1}^{\infty}{{1}\over{4n^{1.5}}}\ $ сходится (см. $\S$1), тогда по следствию выше п.3 наш ряд тоже сходится.}
        }
        \vspace{0.6cm}
        \addcontentsline{toc}{subsubsection}{Теорема. Признак Коши}
        \par{\textbf{Теорема. "Признак Коши"}\\
        \text{Пусть $a_n\geq 0$ и $\displaystyle\exists\lim_{n\rightarrow+\infty}\sqrt[n]{a_n}=q$. Тогда возможны три случая:}}
        \begin{itemize}
            \item \par{если q < 1, то $\displaystyle\sum_{n=1}^{\infty}a_n\ $ сходится;}
            \item \par{если q > 1, то $\displaystyle\sum_{n=1}^{\infty}a_n\ $ расходится;}
            \item \par{если q = 1, то про сходимость $\displaystyle\sum_{n=1}^{\infty}a_n\ $ ничего сказать нельзя;}
        \end{itemize}
        \textbf{Доказательство:}
        \begin{itemize}
            \item \par{
            \text{[q<1] Возьмём $\epsilon>0$, т.ч. $q+\epsilon<1$, тогда, начиная с некоторого номера,}\\
            \text{$\sqrt[n]{a_n}<q+\epsilon$ (правая часть определения предела) $\Rightarrow 0\leq a_n<(q+\epsilon)^n$, т.к. $q+\epsilon<1$, то}\\
            \text{ряд $\displaystyle\sum_{n=1}^{\infty}a_n\ $ сходится по признаку сравнения ($\displaystyle\sum_{n=1}^{\infty}(q+\epsilon)^n \ $ сходится при $q+\epsilon<1$).}
            }
            \item \par{
            \text{[q>1] Возьмём $\epsilon>0$, т.ч. $q-\epsilon>1$, тогда, начиная с некоторого номера,}\\
            \text{$q-\epsilon<\sqrt[n]{a_n}$ (левая часть определения предела) $\Rightarrow (q-\epsilon)^n<a_n$, т.к. $q-\epsilon>1$, то}\\
            \text{ряд $\displaystyle\sum_{n=1}^{\infty}a_n\ $ расходится по признаку сравнения ($\displaystyle\sum_{n=1}^{\infty}(q-\epsilon)^n \ $ расходится при $q-\epsilon>1$).}
            }
            \item \par{
            \text{[q=1] Рассмотрим $a_n={{1}\over{n^p}}$. $\displaystyle\sqrt[n]{{1}\over{n^p}}={{1}\over{n^{{p}\over{n}}}}\xrightarrow{n\rightarrow+\infty}1\ $, но при разных p разная сходимость.}
            }
        \end{itemize}
        \par{{\raggedleft ч.т.д.\par}}
        \newpage
        \addcontentsline{toc}{subsubsection}{Теорема. Признак Даламбера}
        \par{\textbf{Теорема. "Признак Даламбера"}\\
        \text{Пусть $a_n>0$ и $\displaystyle\exists\lim_{n\rightarrow+\infty}{{a_{n+1}}\over{a_n}}=q$. Тогда возможны три случая:}}
        \begin{itemize}
            \item \par{если q < 1, то $\displaystyle\sum_{n=1}^{\infty}a_n\ $ сходится;}
            \item \par{если q > 1, то $\displaystyle\sum_{n=1}^{\infty}a_n\ $ расходится;}
            \item \par{если q = 1, то про сходимость $\displaystyle\sum_{n=1}^{\infty}a_n\ $ ничего сказать нельзя;}
        \end{itemize}
        \textbf{Доказательство:}
        \begin{itemize}
            \item \par{
            \text{[q<1] Возьмём $\epsilon>0$, т.ч. $q+\epsilon<1$, тогда, начиная с некоторого номера $N_0\in\mathbb{N}$,}\\
            \text{$\displaystyle{{a_{n+1}}\over{a_n}}<q+\epsilon$ (правая часть определения предела).}\\
            \text{Откуда $\displaystyle a_n<(q+\epsilon)a_{n-1}<(q+\epsilon)^2a_{n-2}<...<(q+\epsilon)^{n-N_0}a_{N_0}$. Таким образом} \\
            \text{получаем, что $\displaystyle a_n<\underbrace{(q+\epsilon)^{-N_0}a_{N_0}}_{const=c}(q+\epsilon)^n$. Если $q+\epsilon<1$, то $\displaystyle c\sum_{n=1}^{\infty}(q+\epsilon)^n\ $ сходится.}\\
            \text{Тогда по признаку сравнения наш ряд тоже сходится.}
            }
            \item \par{
            \text{[q>1] Возьмём $\epsilon>0$, т.ч. $q-\epsilon>1$, тогда, начиная с некоторого номера $N_0\in\mathbb{N}$,}\\
            \text{$\displaystyle q-\epsilon <{{a_{n+1}}\over{a_n}}$ (левая часть определения предела).}\\
            \text{Откуда $\displaystyle a_n>(q-\epsilon)a_{n-1}>(q-\epsilon)^2a_{n-2}>...>(q-\epsilon)^{n-N_0}a_{N_0}$. Таким образом} \\
            \text{получаем, что $\displaystyle a_n>\underbrace{(q-\epsilon)^{-N_0}a_{N_0}}_{const=c}(q-\epsilon)^n$. Если $q-\epsilon>1$, то $\displaystyle c\sum_{n=1}^{\infty}(q-\epsilon)^n\ $ расходится.}\\
            \text{Тогда по признаку сравнения наш ряд тоже расходится.}
            }
            \item \par{
            \text{[q=1] Рассмотрим $a_n={{1}\over{n^p}}$. $\displaystyle{{a_{n+1}}\over{a_n}}=({{n}\over{n+1}})^p\xrightarrow{n\rightarrow+\infty}1\ $, но при}\\\text{разных p разная сходимость.}
            }
        \end{itemize}
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{1cm}
        \par{
        \textbf{Лемма 1. }\text{Пусть $\displaystyle M=\lim_{k\rightarrow\infty}\underbrace{sup\{a_k,a_{k+1},...\}}_{=M_k}$.
        $\forall\epsilon>0 \ \ \forall N\in\mathbb{N}$  $\exists n>N: \ \ a_n>M-\epsilon$} \\
        \textbf{Доказательство: }\text{Фикс. $\epsilon>0$ и $N\in\mathbb{N}$, т.к. $M_k$ убывает к $M$, то $M_k\geq M$ $\forall k\in\mathbb{N}$.}\\
        \text{Пусть $k>N$. При этом по орпеделению $sup$ $\exists n\geq k>N$, т.ч. $a_n>M_k-\epsilon \Rightarrow$}\\\text{$\Rightarrow a_n>M_k-\epsilon\geq M-\epsilon$. (данная лемма из 1-ого семестра)}
        }
        \par{{\raggedleft ч.т.д.\par}}

        \addcontentsline{toc}{subsubsection}{Теорема. Обобщённый признак Коши}
        \par{\textbf{Теорема. "Обобщённый признак Коши"}\\
        \text{Пусть $a_n\geq0$ и $\displaystyle\exists\overline{\lim_{n\rightarrow\infty}}\sqrt[n]{a_n}=q$. Тогда возможны три случая:}}
        \begin{itemize}
            \item \par{если q < 1, то $\displaystyle\sum_{n=1}^{\infty}a_n\ $ сходится;}
            \item \par{если q > 1, то $\displaystyle\sum_{n=1}^{\infty}a_n\ $ расходится;}
            \item \par{если q = 1, то про сходимость $\displaystyle\sum_{n=1}^{\infty}a_n\ $ ничего сказать нельзя;}
        \end{itemize}
        \textbf{Доказательство:}
        \begin{itemize}
            \item \par{
            \text{[q<1] Возьмём $\epsilon>0$, т.ч. $q+\epsilon<1$. Т.к. $\displaystyle\overline{\lim_{n\rightarrow\infty}}\sqrt[n]{a_n}=\limsup{\{\sqrt[n]{a_n},\sqrt[n+1]{a_{n+1}},...\}}$, то,}\\
            \text{начиная с некоторого номера, $\sup\{\sqrt[n]{a_n},\sqrt[n+1]{a_{n+1}},...\}<q+\epsilon\Rightarrow\sqrt[n]{a_n}\leq\sup\{...\}<q+\epsilon$.}\\
            \text{Т.е. $a_n<(q+\epsilon)^n$. Т.к. $q+\epsilon<1$, то по признаку сравнения $\displaystyle\sum_{n=1}^{\infty}a_n\ $ сходится. }
            }
            \item \par{
            \text{[q>1] Возьмём $\epsilon>0$, т.ч. $q-\epsilon>1$. Т.к. $\displaystyle\overline{\lim_{n\rightarrow\infty}}\sqrt[n]{a_n}=\limsup{\{\sqrt[n]{a_n},\sqrt[n+1]{a_{n+1}},...\}}$, то  }\\
            \text{по лемме 1 (см. выше) $\forall N\in\mathbb{N}$ $\exists k>N$, т.ч. $\sqrt[k]{a_k}>q-\epsilon$. Т.о. строим $\{n_k\}$, что}\\
            \text{$\displaystyle a_{n_k}>(q-\epsilon)^{n_k}$, при этом $\displaystyle q-\epsilon>1\Rightarrow a_{n_k}\xrightarrow{n_k\rightarrow\infty}\infty$. Если ряд сходится, то общий}\\
            \text{член ряда стремится к 0. Поэтому наш ряд расходится.}
            }
            
        \end{itemize}
        \par{{\raggedleft ч.т.д.\par}}
        \text{!Если сработал признак Даламбера, то и сработал признак Коши!}\\\text{ОБРАТНОЕ НЕВЕРНО!}
        \par{\textbf{Примеры:}}
        \begin{enumerate}
            \item \par{
            \begin{equation*}
                \sum_{n=1}^{\infty}\frac{3+(-1)^n}{2^n}. \ \ \frac{a_{n+1}}{a_n}=\frac{1}{2}*\frac{3+(-1)^{n+1}}{3+(-1)^n}= \begin{cases}
                    {1\over{4}},\text{n - чётное}\\
                    {1},\text{n - нечётное}
                \end{cases} \Rightarrow \text{нет предела.}
            \end{equation*}
            \begin{equation*}
                \sqrt[n]{a_n}={\sqrt[n]{3+(-1)^n}\over{2}}\rightarrow{1\over{2}} \text{ - из признака Коши.}
            \end{equation*}
            }
            \item \par{
            \begin{equation*}
                a+ab+a^2b+a^2b^2+...+a^nb^{n-1}+a^nb^n+..., \ a>0, \ b>0.
            \end{equation*}
            \begin{equation*}
                a_n=\begin{cases}
                    a^kb^{k-1}, \ n=2k-1 \\
                    a^kb^k \ \ \ , \ n = 2k
                \end{cases}
            \end{equation*}
            \begin{equation*}
                \sqrt[2k]{a^kb^k}=\sqrt{ab}, \ n=2k
                \end{equation*}
                \begin{equation*}
                    \sqrt[2k-1]{a^kb^{k-1}}=\sqrt[2k-1]{a^{k-{1\over{2}}}b^{k-{1\over{2}}}a^{1\over{2}}b^{-{1\over{2}}}}=\sqrt{ab}\sqrt[2n]{a\over{b}}, \ n=2k-1
            \end{equation*}
            \begin{equation*}
                \sqrt[n]{a_n}=\begin{cases} 
                \sqrt{ab} \ \ \ \ \ \ , \ n=2k \\
                \sqrt{ab}\sqrt[2n]{a\over{b}}, \ n=2k-1
                \end{cases}
                \xrightarrow{n\rightarrow\infty}\sqrt{ab}
            \end{equation*}
            \text{Если ab=1, то $a+ab+a^2b+a^2b^2+...=a+ab+ab(a)+(ab)^2+...=$} \\
            \text{$=a+1+a+1+\ldots$ - расходится.}}
        \end{enumerate}

        \addcontentsline{toc}{subsubsection}{Теорема. Интегральный признак сходимости ряда}
        \par{\textbf{Теорема. Интегральный признак сходимости ряда} \\
        \text{Пусть f - монотонно убывающая положительная функция на [1, +$\infty$), тогда ряд} \\
        \text{$\displaystyle\sum_{n=1}^{\infty}f(n)$ сходится или расходится одновременно с интегралом $\displaystyle\int_{1}^{+\infty}f(x)dx$.}\\
        \textbf{Доказательство:} \text{т.к. $f(x)$ монотонна, то она интегрируема на $\forall$ конечном интервале.} \\
        \textbf{[$\Rightarrow$]}\text{из сходимости ряда докажем сходимость интеграла}\\
        \text{$\displaystyle f(n+1)\leq f(x)\leq f(n)\Rightarrow\int_{n}^{n+1}f(n+1)dx\leq\int_{n}^{n+1}f(x)dx\leq\int_{n}^{n+1}f(n)dx$}
        \begin{equation*}
            \int_{n}^{n+1}\underbrace{f(n+1)}_{const}dx=f(n+1)\int_{n}^{n+1}dx=f(n+1)*(n+1-n)=f(n+1)
        \end{equation*}
        \text{Из аналогичных соображений получаем, что $\displaystyle\int_{n}^{n+1}f(n)dx=f(n)$.}\\
        \text{$\displaystyle\int_{n}^{n+1}f(n+1)dx\leq\int_{n}^{n+1}f(x)dx\leq\int_{n}^{n+1}f(n)dx\Rightarrow f(n+1)\leq\int_{n}^{n+1}f(x)dx\leq f(n)$.}\\
        \text{Просуммируем по n = 1, 2, ..., N:}
        \begin{equation*}
            \sum_{n=1}^Nf(n+1)\leq\sum_{n=1}^{N}\int_{n}^{n+1}f(x)dx\leq\sum_{n=1}^{N}f(n)\Rightarrow f(2)+...+f(N+1)\leq\underbrace{\int_{1}^{N+1}f(x)dx}_{\text{аддитивность}}\leq f(1)+...+f(N)
        \end{equation*}
        \text{$f(2)+...+f(N+1)=S_{N+1}-f(1)$, $f(1)+...+f(N)=S_N$}
        \begin{equation}
            \label{1}
            S_{N+1}-f(1)\leq\int_{1}^{N+1}f(x)dx\leq S_N
        \end{equation}
        \text{Пусть $\displaystyle\sum_{n=1}^{\infty}f(n)$ сходится, тогда $S_N\leq c$ ограничена $\displaystyle\Rightarrow\int_{1}^{N+1}f(x)dx\leq S_N\leq c\Rightarrow$} \\
        \text{последовательность $\displaystyle\int_{1}^{N+1}f(x)dx$ возрастает и ограничена $\displaystyle\Rightarrow\exists\lim_{N\rightarrow\infty}\int_{1}^{N}f(x)dx\Rightarrow$}\\
        \vspace{0.3cm}
        \text{$\Rightarrow$ наш интеграл сходится.}\\
        \textbf{[$\Leftarrow$]}\text{из сходимости интеграла докажем скодимость ряда}\\
        \text{В предыдущем пункте в формуле $\ref{1}$ заменим N+1 на N: $\displaystyle S_N-f(1)\leq\int_{1}^{N}f(x)dx\Rightarrow$}\\
        \text{$\displaystyle\Rightarrow S_N\leq f(1)+\int_{1}^{N}f(x)dx\leq\int_{1}^{+\infty}f(x)dx + f(1)=C$, т.е. $S_N\leq C$ ограничена, а по лемме}\\
        \text{Критерий сходимости ряда из $\S$2 $S_N$ ограничена $\Leftrightarrow$ ряд сходится, т.е. $\displaystyle\sum_{n=1}^{\infty}f(n)$ сходится.}}
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.5cm}
        \par{\textbf{Примеры:}}
        \begin{enumerate}
            \item \par{
            \text{$\displaystyle\zeta(p)=\sum_{n=1}^{\infty}{1\over{n^p}}$, p>0 $\displaystyle f(x)={1\over{x^p}}\downarrow0\sim\int_{1}^{+\infty}{dx\over{x^p}} $ - }
            $
            \begin{cases}
                \text{сходится при $p > 1$}\\
                \text{расходится при $p \leq1$}
            \end{cases}$
         }
         \item \par{
         \text{$f(x)=$}$
         \begin{cases}
             \text{${1\over{x^2}}, \ x\notin\mathbb{N}$}\\
             \text{${1\over{n}}, \ x\in\mathbb{N}$}
         \end{cases}$
         \text{ - пример, почему функция должна быть монотонна.}\\
         \text{$\displaystyle\sum_{n=1}^{\infty}f(n)=\sum_{n=1}^{\infty}{1\over{n}}$ - расходится, а $\displaystyle\int_{1}^{+\infty}f(x)dx=\int_{1}^{+\infty}{dx\over{x^2}}$ - сходится.}
         }
         \newpage
         \item \par{
         \text{$\displaystyle\zeta(p)=\sum_{n=1}^{\infty}{1\over{n^p}}$, $p\rightarrow1$}\\
         \text{$\displaystyle\sum_{n=2}^{N+1}f(n)\leq\int_{1}^{+\infty}f(x)dx\leq\sum_{n=1}^{N}f(n)$, из формулы \ref{1} получаем следующее неравенство:}
         \begin{equation}
             \int_{1}^{N+1}f(x)dx\leq\sum_{n=1}^{N+1}f(n)\leq\int_{1}^{N+1}f(x)dx+f(1)
         \end{equation}
         \text{Подставив в (2) функцию, приняв $p=1+z, \ z\rightarrow0, \ N\rightarrow+\infty$ получим:}
         \begin{equation}
             \int_{1}^{+\infty}{dx\over{x^{1+z}}}\leq\zeta(1+z)\leq\int_{1}^{+\infty}{dx\over{x^{1+z}}}+1
         \end{equation}
         \begin{equation}
             \int_{1}^{+\infty}{dx\over{x^{1+z}}}=\frac{x^{-z}}{-z}\Bigg|_{1}^{+\infty}=0-{1\over{-z}}={1\over{z}}
         \end{equation}
         \text{Подставим (4) в (3) и получим, что $\displaystyle{1\over{z}}\leq\zeta(1+z)\leq{1\over{z}}+1\Rightarrow1\leq z\zeta(1+z)\leq1+z$.}\\
         \text{При $z\rightarrow0$ по теореме О двух милиционерах $z\zeta(1+z)\rightarrow1$, т.е. $\zeta(1+z)={1\over{z}}+\gamma+o(1)$.}
         }
        \end{enumerate}

        \subsection*{1. Знакочередующийся ряд}
        \addcontentsline{toc}{subsection}{1. Знакочередующийся ряд}
        \par{\textbf{Def 1: } \text{$\displaystyle\sum_{n=1}^{\infty}(-1)^{n-1}b_n$, где $b_n>0$ - знакочередующийся ряд.}
        }
        \vspace{0.6cm}
        \addcontentsline{toc}{subsubsection}{Теорема. Признак Лейбница}
        \par{
        \textbf{Теорема. Признак Лейбница} \\
        \text{Пусть задана $b_n$, т.ч. $b_n>b_{n+1}, \ \displaystyle\lim_{n\rightarrow\infty}b_n=0$. Тогда ряд $\displaystyle\sum_{n=1}^{\infty}(-1)^{n-1}b_n$ сходится и}\\\text{его сумма S подчиняется неравенству $0<S<b_1$.} \\
        \textbf{Доказательство: }\text{рассмотрим частичные суммы $S_n$ с n=2k номерами:}
        \begin{equation}
            S_{2N}=\underbrace{(b_1-b_2)}_{>0}+...+\underbrace{(b_{2N-1}-b_{2N})}_{>0}\Rightarrow S_{2N}<S_{2(N+1)}
        \end{equation}
        \begin{equation}
            S_{2N}=b_1-\underbrace{(b_2-b_3)}_{>0}-...\underbrace{(b_{2N-2}-b_{2N-1})}_{>0}-\underbrace{b_{2N}}_{>0}<b_1
        \end{equation}
        \text{Из (1) делаем вывод, что $S_{2N}$ монотонно возрастает, а из (2) делаем вывод, что}\\
        \text{она ограничена. Значит $\displaystyle\exists\lim_{N\rightarrow\infty}S_{2N}=S$.}
        \text{Рассмотрим $\displaystyle S_{2N+1}=\underbrace{S_{2N}}_{\rightarrow S}+\underbrace{b_{2N+1}}_{\rightarrow0}\rightarrow S$:}\\
        \text{Значит $\displaystyle\exists\lim_{N\rightarrow\infty}S_N=S$ и $\displaystyle\sum_{n=1}^{\infty}(-1)^{n-1}b_n$ сходится.} 
        \begin{equation}
            0<S_2<S_{2N}<S \text{, где $S_2$ - сумма первых 2-х элементов ряда}
        \end{equation}
        \text{Посмотрим на $S_{2N+1}=b_1-(b_2-b_3)-...-(b_{2N}-b_{2N+1})\Rightarrow S_{2N+1}$ монотонно убывает, }\\
        \newpage
        \text{при этом недавно выяснили, что она стремится к S $\Rightarrow S<S_{2N+1}\Rightarrow$:}
        \begin{equation}
            S<S_1=b_1
        \end{equation}
        \text{Объединив (3) и (4), получаем, что $0<S<b_1$.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.5cm}
        \par{\textbf{Примеры:}}
        \begin{enumerate}
            \item \par{
            \textbf{Ряд Лейбница} \text{: $\displaystyle\sum_{n=1}^{\infty}\frac{(-1)^{n-1}}{n}$ сходится, т.к. $b_n = {1\over{n}}$ монотонно идёт к 0.}
            }
            \item \par{
            \text{пример, почему нельзя избавиться от монотонности: $\displaystyle\sum_{n=1}^{\infty}(-1)^{n-1}\frac{2+(-1)^{n}}{n}$}
            \begin{equation*}
                \sum_{n=1}^{\infty}(-1)^{n-1}\frac{2+(-1)^{n}}{n}=2\underbrace{\sum_{n=1}^{\infty}\frac{(-1)^{n-1}}{n}}_{\text{сходится}}-\underbrace{\sum_{n=1}^{\infty}\frac{1}{n}}_{\text{расходится}}
            \end{equation*}
            \text{$\Rightarrow$ и наш ряд тоже расходится.}
            }
        \end{enumerate}

        \subsection*{2. Признаки Абеля и Дирихле}
        \addcontentsline{toc}{subsection}{2. Признаки Абеля и Дирихле}
        \setcounter{equation}{0}
        \par{
        \text{Рассмотрим $\displaystyle\sum_{n=1}^{\infty}a_nb_n$. Обозначим за $\displaystyle B_n=\sum_{n=1}^{N}b_n$, тогда $b_n=B_n-B_{n-1}$.}\\
        \text{При этом $\displaystyle S_N=\sum_{n=1}^{N}a_nb_n=a_1B_1+a_2(B_2-B_1)+...+a_N(B_N-B_{N-1})$, тогда получим:}
        \begin{equation}
            \sum_{n=1}^{N}a_nb_n=a_NB_N+\sum_{n=1}^{N-1}(a_n-a_{n+1})B_n \text{ - суммирование по частям.}
        \end{equation}
        }
        \vspace{0.5cm}
        \addcontentsline{toc}{subsubsection}{Теорема. Признак Абеля}
        \par{
        \textbf{Теорема. Признак Абеля}\\
        \text{Пусть $\displaystyle\sum_{n=1}^{\infty}b_n$ сходится, а последовательность $\{a_n\}$ монотонна и ограничена, тогда}\\
        \text{ряд $\displaystyle\sum_{n=1}^{\infty}a_nb_n$ сходится.}\\
        \textbf{Доказательство: }\text{распишем критерий Коши: фикс. $\epsilon>0$, т.к. $\displaystyle\sum_{n=1}^{\infty}b_n$ сходится, то}\\
        \text{$\exists N\in\mathbb{N}: \ \forall n>N, \ p\in\mathbb{N} \ \ |b_{n+1}+\ldots+b_{n+p}|=|\tilde{B}_p|<\epsilon$.}\\
        \text{Фикс. $n>N$, тогда для $\displaystyle\tilde{B}_p=\sum_{m=1}^pb_{n+m}$ верно, что $|\tilde{B}_p|<\epsilon$. Т.к. $a_n$ ограничена, то}\\
        \text{$\exists c>0: \ |a_n|\leq c \ \ \ \forall n\in\mathbb{N}$.}
        \newpage
        \begin{equation*}
            |\sum_{q=1}^pa_{n+q}b_{n+q}|=|a_{n+p}\tilde{B}_p+\sum_{q=1}^{p-1}(a_{n+q}-a_{n+q+1})\tilde{B}_q\leq|a_{n+p}|\underbrace{|\tilde{B}_p|}_{<\epsilon}+\sum_{q=1}^{p-1}(|a_{n+q}-a_{n+q+1}|\underbrace{|\tilde{B}_q|}_{<\epsilon}) <
        \end{equation*}
        \begin{equation}
            <\epsilon(|a_{n+p}|+\sum_{q=1}^{p-1}|a_{n+q}-a_{n+q+1}|)
        \end{equation}
        \text{Предположим, что $a_n$ монотонно убывает, тогда (2) можно переписать ввиде:}
        \begin{equation}
            \epsilon(|a_{n+p}|+a_{n+1}-a_{n+2}+a_{n+2}-a_{n+3}+\ldots-a_{n+p})=\epsilon(|a_{n+p}|+\underbrace{(a_{n+1}-a_{n+p})}_{>0})
        \end{equation}
        \text{Предположим, что $a_n$ монотонно возрастает, тогда (2) можно переписать ввиде:}
        \begin{equation}
            \epsilon(|a_{n+p}|-a_{n+1}+a_{n+2}-a_{n+2}+a_{n+3}-\ldots+a_{n+p})=\epsilon(|a_{n+p}|-\underbrace{(a_{n+1}-a_{n+p})}_{<0})
        \end{equation}
        \text{Заметим, что в (3) и (4) $a_{n+1}-a_{n+p}$ стоят с разными знаками, значит в итоговой}\\\text{форме мы напишем просто модуль:}
        \begin{equation}
            (2) = \epsilon(|a_{n+p}|+|a_{n+1}-a_{n+p}|)\leq\epsilon(\underbrace{|a_{n+p}|}_{<c}+\underbrace{|a_{n+1}|}_{<c}+\underbrace{|a_{n+p}|}_{<c})\leq3c\epsilon
        \end{equation}
        \text{Из (5) следует, что $\tilde{B}_p$ можно сделать сколь угодно малым $\forall p\in\mathbb{N}$. Тогда по} \\
        \text{критерию Коши ряд $\displaystyle\sum_{n=1}^{\infty}a_nb_n$ сходится.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.7cm}
        \addcontentsline{toc}{subsubsection}{Теорема. Признак Дирихле}
        \setcounter{equation}{0}
        \par{
        \textbf{Теорема. Признак Дирихле}\\
        \text{Пусть $\displaystyle B_N=\sum_{n=1}^Nb_n$ (последовательность частичных сумм) ограничена, а последова-}\\\text{тельность $\{a_n\}$ монотонно стремится к нулю. Тогда ряд $\displaystyle\sum_{n=1}^{\infty}a_nb_n$ сходится.}\\
        \textbf{Доказательство: } \text{фикс. $\epsilon>0$, т.к. $a_n\rightarrow0$, то $\exists N\in\mathbb{N}$, т.ч. $\forall n>N \ \ |a_n|<\epsilon$.} \\
        \text{Т.к. $B_N$ ограничена $|B_N|\leq c \ \ \forall N\in\mathbb{N}$, то $|\tilde{B}_p|=|B_{n+p}-B_n|\leq|B_{n+p}|+|B_n|\leq2c$.}\\
        \text{Фикс. $n\geq N$, тогда}
        \begin{equation*}
            |\sum_{q=1}^pa_{n+q}b_{n+q}|=|a_{n+p}\tilde{B}_p+\sum_{q=1}^{p-1}(a_{n+q}-a_{n+q+1})\tilde{B}_q|\leq|a_{n+p}|\underbrace{|\tilde{B}_p|}_{<2c}+\sum_{q=1}^{p-1}(|a_{n+q}-a_{n+q+1}|\underbrace{|\tilde{B}_q|}_{2c}) <
        \end{equation*}
        \begin{equation}
            <2c(|a_{n+p}|+\sum_{q=1}^{p-1}|a_{n+q}-a_{n+q+1}|)
        \end{equation}
        \text{Предположим, что $a_n$ монотонно убывает, тогда (1) можно переписать ввиде:}
        \begin{equation}
            2c(|a_{n+p}|+a_{n+1}-a_{n+2}+a_{n+2}-a_{n+3}+\ldots-a_{n+p})=2c(|a_{n+p}|+\underbrace{(a_{n+1}-a_{n+p})}_{>0})
        \end{equation}
        \newpage
        \text{Предположим, что $a_n$ монотонно возрастает, тогда (2) можно переписать ввиде:}
        \begin{equation}
            2c(|a_{n+p}|-a_{n+1}+a_{n+2}-a_{n+2}+a_{n+3}-\ldots+a_{n+p})=2c(|a_{n+p}|-\underbrace{(a_{n+1}-a_{n+p})}_{<0})
        \end{equation}
        \text{Заметим, что в (2) и (3) $a_{n+1}-a_{n+p}$ стоят с разными знаками, значит в итоговой}\\\text{форме мы напишем просто модуль:}
        \begin{equation}
            (1) = 2c(|a_{n+p}|+|a_{n+1}-a_{n+p}|)\leq2c(\underbrace{|a_{n+p}|}_{<\epsilon}+\underbrace{|a_{n+1}|}_{<\epsilon}+\underbrace{|a_{n+p}|}_{<\epsilon})\leq6c\epsilon
        \end{equation}
        \text{Из (4) следует, что $\tilde{B}_p$ можно сделать сколь угодно малым $\forall p\in\mathbb{N}$. Тогда по} \\
        \text{критерию Коши ряд $\displaystyle\sum_{n=1}^{\infty}a_nb_n$ сходится.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.5cm}
        \par{
        \textbf{Пример:}
        }
        \par{\text{$\displaystyle\sum_{n=1}^{\infty}\frac{\sin{nx}}{n^p}, \ p>0. \ {1\over{n^p}}\downarrow0 $ и $|\sin{nx}|<1$. Тогда $\displaystyle B_N=\sum_{n=1}^N\sin{nx}=\frac{\sin{Nx\over{2}}\sin{{N+1\over2}x}}{\sin{x\over2}}$.}\\
        \text{Если $\sin{x\over2}=0$, то $\sin{nx}=0$ и все члены ряда тоже 0, т.е. ряд сходится. Иначе} \\
        \text{$\displaystyle|B_N|=|\sum_{n=1}^N\sin{nx}|=|\frac{\sin{Nx\over{2}}\sin{{N+1\over2}x}}{\sin{x\over2}}|\leq{1\over{\sin{x\over2}}}$ - ограничена.}
        \begin{equation*}
            \sum_{n=1}^N\sin{nx}=\sum_{n=1}^N{(e^{ix}-e^{-ix})\over{2i}}={1\over{2i}}(\sum_{n=1}^Ne^{ix}-\sum_{n=1}^Ne^{-ix})={1\over{2i}}(\frac{e^{ix}(e^{iNx}-1)}{e^{ix}-1}-\frac{e^{-ix}(e^{-iNx}-1)}{e^{-ix}-1})=
        \end{equation*}
        \begin{equation*}
            ={1\over{2i}}(\frac{e^{ix\over2}(e^{iNx}-1)
                }{e^{ix\over2}-e^{-ix\over2}}+\frac{e^{-ix\over{2}}(e^{-iNx}-1)}{e^{ix\over2}-e^{-ix\over2}})={1\over{2i}}\frac{(e^{i(N+{1\over{2}})x}+e^{-i(N+{1\over{2}})x})-(e^{i{1\over{2}}x}+e^{-i{1\over{2}}x})}{e^{ix\over2}-e^{-ix\over2}}=
        \end{equation*}
        \begin{equation*}
            =-\frac{(e^{i(N+{1\over{2}})x}+e^{-i(N+{1\over{2}})x})-(e^{i{1\over{2}}x}+e^{-i{1\over{2}}x})}{4\sin{x\over{2}}}=-\frac{\cos{(N+{1\over2})x}-\cos{{1\over2}x}}{2\sin{x\over2}}=\frac{\sin{Nx\over{2}}\sin{{N+1\over2}x}}{\sin{x\over2}}
        \end{equation*}
        }
        \vspace{0.7cm}
        \subsection*{3. Абсолютная и условная сходимость}
        \addcontentsline{toc}{subsection}{3. Абсолютная и условная сходимость}
        \par{
        \textbf{Лемма 1. } \text{Пусть ряд $\displaystyle\sum_{n=1}^{\infty}|a_n|$ сходится, тогда и ряд $\displaystyle\sum_{n=1}^{\infty}a_n$ сходится.} \\
        \textbf{Доказательство: } \text{применим критрерий Коши: из сходимости ряда $\displaystyle\sum_{n=1}^{\infty}|a_n|$:} \\
        \text{$\displaystyle\forall\epsilon>0 \ \ \ \exists N\in\mathbb{N}$: $\forall n>N$, $p\in\mathbb{N}$ $|a_{n+1}|+\ldots+|a_{n+p}|<\epsilon$.} \\
        \text{При тех же $N$ и $p$ $|a_{n+1}+\ldots+a_{n+p}|\leq|a_{n+1}|+\ldots+|a_{n+p}|<\epsilon\Rightarrow\displaystyle\sum_{n=1}^{\infty}a_n$ сходится} \\
        \text{по критерию Коши.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.5cm}
        \par{
        \textbf{Def 1: }\text{Ряд $\displaystyle\sum_{n=1}^{\infty}a_n$ называется абсолютно сходящимся, если ряд $\displaystyle\sum_{n=1}^{\infty}|a_n|$ сходится.}\\
        \text{По лемме выше абсолютно сходящийся ряд сходится.}
        }
        \newpage
        \par{
        \textbf{Def 2: }\text{Ряд $\displaystyle\sum_{n=1}^{\infty}a_n$ называется условно сходящимся, если ряд $\displaystyle\sum_{n=1}^{\infty}|a_n|$ расходится,}\\
        \text{но сам ряд $\displaystyle\sum_{n=1}^{\infty}a_n$ сходится.}
        }
        \vspace{0.5cm}
        \par{
        \textbf{Пример: }\text{Ряд $\displaystyle\sum_{n=1}^{\infty}\frac{(-1)^{n-1}}{n}$ - сходится, а $\displaystyle\sum_{n=1}^{\infty}\frac{1}{n}$ расходится.}
        }
        \par{\text{Рассмотрим произвольный ряд $\displaystyle\sum_{n=1}^{\infty}a_n$. Обозначим за $p_k$ неотрицательные члены,} \\
        \text{а за $-q_m$ - отрицательные. При этом нумерация $-q_m$ и $p_k$ не зависит от нумерации ряда. }\\
        \text{Например, первые два неотрицательных члена в $a_n: a_{2}$ и $a_{8}$. Тогда $p_1=a_2$, $p_2=a_8$.} \\
        \text{Т.е. нумерация происходит по мере их появления.}
        }
        \vspace{0.6cm}
        \setcounter{equation}{0}
        \par{
        \textbf{Лемма 2.} \text{Ряд $\displaystyle\sum_{n=1}^{\infty}a_n$ сходится абсолютно $\Leftrightarrow$ оба ряда $\displaystyle\sum_{k}p_k$ и $\displaystyle\sum_{m}q_m$ сходятся.} \\
        \text{Важно отметить, что при этом не исключён случай, когда один из рядов конечен} \\
        \text{или вовсе отсутствует.} \\
        \textbf{Доказательство: } \text{$S_N=\displaystyle\sum_{n=1}^{N}a_n=\displaystyle\sum_{k=1}^{K}p_k-\displaystyle\sum_{m=1}^{M}q_m$, $N=K+M$. Тогда получим:}\\
        \begin{equation}
            S_N:=P_K-Q_M
        \end{equation}
        \text{$\tilde{S}_N=\displaystyle\sum_{n=1}^{N}|a_n|=\displaystyle\sum_{k=1}^{K}p_k+\displaystyle\sum_{m=1}^{M}q_m=P_K+Q_M.$ Тогда получим следующее равенство:}
        \begin{equation}
            \tilde{S}_N=P_K+Q_M
        \end{equation}
        \text{Сумма (1) + (2) даёт нам $P_K={1\over2}(\tilde{S}_N+S_N)$, а разность (2) - (1) - $Q_M={1\over2}(\tilde{S}_N-S_N)$.} \\
        \textbf{[$\Rightarrow$]}\text{из абсолютной сходимости ряда докажем сходимость $\displaystyle\sum_{k}p_k$ и $\displaystyle\sum_{m}q_m$:}\\
        \text{Действительно, если ряд $\displaystyle\sum_{n=1}^{\infty}a_n$ абсолютно сходится, то $\exists$ конечные $\displaystyle\lim_{N\rightarrow\infty}S_N$ и $\displaystyle\lim_{N\rightarrow\infty}\tilde{S}_N$.}\\
        \text{Тогда существуют и конечные пределы $P_K={1\over2}(\tilde{S}_N+S_N)$ и $Q_M={1\over2}(\tilde{S}_N-S_N)$.}\\
        \textbf{[$\Leftarrow$]}\text{из сходимости $\displaystyle\sum_{k}p_k$ и $\displaystyle\sum_{m}q_m$ докажем сходимость $\displaystyle\sum_{n=1}^{\infty}a_n$:}\\
        \text{Если в них бесконечное кол-во элементов, то $\exists$ их конечные пределы. Тогда по}\\
        \text{формулам (1) и (2) $\exists$ конечные $\displaystyle\lim_{N\rightarrow\infty}S_N$ и $\displaystyle\lim_{N\rightarrow\infty}\tilde{S}_N$. Если же один ряд конечен,}\\
        \text{а другой сходится, то тоже всё хорошо. Просто будет фигурировать один из пределов.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.5cm}
        \par{
        \textbf{Лемма 3. } \text{Пусть $\displaystyle\sum_{n=1}^{\infty}a_n$ сходится условно, тогда ряды $\displaystyle\sum_{k}p_k$ и $\displaystyle\sum_{m}q_m$ расходятся.} \\
        \text{Замечание: здесь, в отличие от леммы 2, лишь необходимое условие.}\\
        \textbf{Доказательство: } {$\displaystyle\sum_{k}p_k$ и $\displaystyle\sum_{m}q_m$ точно не конечны. Переберём все другие варианты:}
        }
        \begin{itemize}
            \item \par{
            \text{Пусть $\displaystyle\sum_{k}p_k$ и $\displaystyle\sum_{m}q_m$ сходятся, тогда по лемме 2 наш ряд сходится абсолютно. $\lightning$}
            }
            \newpage
            \item \par{
            \text{Пусть $\displaystyle\sum_{k}p_k$ сходится, а $\displaystyle\sum_{m}q_m$ расходится. Пусть $P=\displaystyle\sum_{k=1}^{\infty}p_k$, тогда}\\
            \text{$S_N=\underbrace{P_K}_{\leq P}- \ Q_M \leq P-Q_M \rightarrow -\infty.$ Но тогда $S_N$ тоже стремится к $-\infty$.}\\
            \text{Тогда $\displaystyle\sum_{n=1}^{\infty}a_n$ расходится, что приводит к противоречию с предположением. $\lightning$}
            }
            \item \par{
            \text{Пусть $\displaystyle\sum_{k}p_k$ расходится, а $\displaystyle\sum_{m}q_m$ сходится обобщается на предыдущий случай}\\
            \text{домножением на -1. По лемме 2 $\S1$ этот ряд тоже расходится. $\lightning$}\\
            }
            \text{Все остальные случаи приводят к противоречию, тогда остаётся единственный}\\
            \text{случай, когда ряды $\displaystyle\sum_{k}p_k$ и $\displaystyle\sum_{m}q_m$ расходятся.}
        \end{itemize}
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.5cm}
        \par{
        \textbf{Теорема 1. }\text{Пусть ряд $\displaystyle\sum_{n=1}^{\infty}a_n$ сходится абсолютно, тогда ряд $\displaystyle\sum_{n=1}^{\infty}\tilde{a}_n$, полученный}\\
        \text{из ряда $\displaystyle\sum_{n=1}^{\infty}a_n$ перестановкой слагаемых, сходится абсолютно и имеет ту же сумму.}\\
        \textbf{Доказательство: }\text{Пусть $\displaystyle\sum_{n=1}^{\infty}a_n$ положителен. Рассмотрим $\tilde{S}_{\tilde{N}}=\displaystyle\sum_{n=1}^{\tilde{N}}\tilde{a}_n$. Где $\tilde{a}_n=a_{f(n)}$,}\\
        \text{где $f(n)$ - некоторая функция перестановки слагаемых. Пусть $N=\max f(n)$, $n=1,2,...,\tilde{N}$.}\\
        \text{Т.е. мы пронумеровали элементы после перестановки по порядку и при помощи функции}\\
        \text{получили порядковые номера элементов, которые были у них до перестановки.}\\
        \text{Тогда $S_N\geq \tilde{S}_{\tilde{N}}$, ведь в $S_N$ содержатся все элементы $\tilde{S}_{\tilde{N}}$. По условию ряд $\displaystyle\sum_{n=1}^{\infty}a_n$ сходится}\\
        \text{абсолютно. Тогда по лемме Критерий сходимости положительного ряда из $\S2$ следует, что}\\
        \text{$S_N$ ограничен, т.е. $S_N\leq C$. Тогда $\tilde{S}_{\tilde{N}}\leq S_N\leq C$, т.е. $\tilde{S}_{\tilde{N}}$ тоже ограничен. По той же лемме}\\
        \text{$\displaystyle\sum_{n=1}^{\infty}\tilde{a}_n$ сходится. Т.к. $\displaystyle\sum_{n=1}^{\infty}\tilde{a}_n$ положителен, то он сходится абсолютно (модуль элемента}\\
        \text{равен элементу). Обозначим за $\displaystyle\tilde{S}=\lim_{\tilde{N}\rightarrow\infty}\tilde{S}_{\tilde{N}}$, тогда $S_N\geq \tilde{S}_{\tilde{N}} \Rightarrow S\geq\tilde{S}$.}\\
        \text{Пересталяя члены ряда $\displaystyle\sum_{n=1}^{\infty}\tilde{a}_n$ вернёмся к ряду $\displaystyle\sum_{n=1}^{\infty}a_n$. По доказанному $S\leq\tilde{S}$. Тогда}\\
        \text{$S\leq\tilde{S}\leq S \Rightarrow S=\tilde{S}$.} \\
        \text{Пусть $\displaystyle\sum_{n=1}^{\infty}a_n$ - произвольный абсолютно сходящийся ряд. Тогда по лемме 2 на предыд.}\\
        \text{странице положительные ряды $\displaystyle\sum_{k}p_k$ и $\displaystyle\sum_{m}q_m$ сходятся. Тогда в силу первой части}\\
        \text{доказательства $\displaystyle\sum_{k}\tilde{p}_k$ и $\displaystyle\sum_{m}\tilde{q}_m$, полученные перестановками слагаемых, сходятся, а их}\\
        \text{суммы не меняются. Тогда в силу леммы 2 с предыдущей страницы $\displaystyle\sum_{n=1}^{\infty}\tilde{a}_n=\displaystyle\sum_{k}\tilde{p}_k-\displaystyle\sum_{m}\tilde{q}_m$}\\
        \text{тоже сходится абсолютно. В данном случае мы не ссылаемся на лемму о сумме рядов, }\\
        \text{но ссылаемся на теорему о переходе к пределу.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \newpage
        \addcontentsline{toc}{subsubsection}{Теорема (Риман)}
        \setcounter{equation}{0}
        \par{
        \textbf{Теорема (Риман) о условно сходящихся рядах.}\\
        \text{Путь $\displaystyle\sum_{n=1}^{\infty}a_n$ сходится условно, тогда перестановкой членов ряда можно добиться,}\\
        \text{чтобы $\displaystyle\sum_{n=1}^{\infty}\tilde{a}_n$ сходился к любому наперёд заданному числу $S$ или вовсе расходился.}\\
        \textbf{Доказательство: } 
        \text{Так как ряд $\displaystyle\sum_{n=1}^{\infty}a_n$ сходится условно, то по лемме выше ряды}\\
        \text{$\displaystyle\sum_{k}p_k$ и $\displaystyle\sum_{m}q_m$ расходятся, т.е. содержат бесконечное кол-во элементов. Будем идти по}\\
        \text{шагам и набирать элементы так, чтобы выполнялись следующие неравенства:}
        \begin{enumerate}
            \item \par{
            \begin{enumerate}
                \item \par{\text{$p_1+p_2+\ldots+p_{k_1-1}\leq S$ - набрали $k_1-1$ положительных элементов}}
                \item \par{\text{$p_1+p_2+\ldots+p_{k_1-1}+p_{k_1}> S$ - набрали $k_1$ый положит. элемент}}
            \end{enumerate}
            }
            \item \par{
            \begin{enumerate}
                \item \par{$p_1+p_2+\ldots+p_{k_1}-q_1-q_2-\ldots-q_{m_1-1}\geq S$
            }
            \item \par{$p_1+p_2+\ldots+p_{k_1}-q_1-q_2-\ldots-q_{m_1-1}-q_{m_1}< S$
            }
            \end{enumerate}
            }
            \item \par{
            \begin{enumerate}
                \item \par{\text{$p_1+\ldots+p_{k_1}+p_{k_1+1}+\ldots+p_{k_2-1}-q_1-\ldots-q_{m_1}\leq S$}}
                \item \par{\text{$p_1+\ldots+p_{k_1}+p_{k_1+1}+\ldots+p_{k_2-1}+p_{k_2}-q_1-\ldots-q_{m_1}> S$}}
            \end{enumerate}
            }
            \item \par{
            \begin{enumerate}
                \item \par{\text{$p_1+\ldots+p_{k_2}-q_1-\ldots-q_{m_1}-q_{m_1+1}-\ldots-q_{m_2-1}\geq S$}}
                \item \par{\text{$p_1+\ldots+p_{k_2}-q_1-\ldots-q_{m_1}-q_{m_1+1}-\ldots-q_{m_2-1}-q_{m_2}< S$}}
            \end{enumerate}
            }
        \end{enumerate}
        \par{\text{$\ldots$}}
        }
        \par{Таким образом на шаге n-1 верны неравенства:}
        \begin{equation*}
            p_1+\ldots+p_{k_{n-1}-1}-q_1-\ldots-q_{m_{n-2}}\leq S
        \end{equation*}
        \begin{equation*}
            p_1+\ldots+p_{k_{n-1}}-q_1-\ldots-q_{m_{n-2}} > S
        \end{equation*}
        \begin{equation}
            p_1+\ldots+p_{k_{n-1}}-q_1-\ldots-q_{m_{n-1}-1}\geq S
        \end{equation}
        \begin{equation}
            \underbrace{p_1+\ldots+p_{k_{n-1}}-q_1-\ldots-q_{m_{n-1}}}_{S_{k_{n-1}+m_{n-1}}}< S
        \end{equation}
        \par{Таким образом на шаге n верны неравенства:}
        \begin{equation}
            p_1+\ldots+p_{k_{n}-1}-q_1-\ldots-q_{m_{n-1}}\leq S
        \end{equation}
        \begin{equation}
            \underbrace{p_1+\ldots+p_{k_{n}}-q_1-\ldots-q_{m_{n-1}}}_{S_{k_{n}+m_{n-1}}} > S
        \end{equation}
        \begin{equation}
            p_1+\ldots+p_{k_{n}}-q_1-\ldots-q_{m_{n}-1}\geq S
        \end{equation}
        \begin{equation*}
            p_1+\ldots+p_{k_{n}}-q_1-\ldots-q_{m_{n}}< S
        \end{equation*}
        \par{\text{Чуточку поясним наши шаги на картинке:}\\
        \includegraphics[width=0.9\linewidth]{28png.png}\\
        \text{Римские цифры 1,2,3,4 обозначают текущее положение нашей суммы. Данными}\newpage
        \text{шагами мы всё время приближаемся к нашей сумме, но промахиваемся мимо.}\\
        \text{Идя т.о. по индукции, мы задействуем весь ряд, ведь на каждом шаге мы добавляем}\\
        \text{хотя бы 1 элемент из нашего ряда, при том чередуем набор + и - элементов ряда.}\\
        }
        \par{\text{Пусть мы не добрали элементов до S, тогда рассмотрим шаг $n$ пункт 1:}\\
        \text{пусть мы набрали N элементов, тогда верно:}}
        \begin{equation}
            \underbrace{k_{n-1}+m_{n-1}}_{\text{на предыд. шаге}}<N\leq \underbrace{k_n-1+m_{n-1}}_{\text{на текущ. шаге}}
        \end{equation}
        \text{Тогда $S_N$ точно больше, чем сумма, полученная на предыдущем шаге (2), т.к.}\\
        \text{элементов мы не добрали и не смогли перескочить S. Тогда:}
        \begin{equation}
            S_{k_{n-1}+m_{n-1}}<S_N
        \end{equation}
        \text{В силу правой части (6) мы находимся не более чем на шаге $n$ пункте 1, а в силу}\\
        \text{этого шага (3):}
        \begin{equation}
            S_N\leq S
        \end{equation}
        \text{Для ограничения $S_{k_{n-1}+m_{n-1}}$ воспользуемся (1): действительно, до $S_{k_{n-1}+m_{n-1}}$ левой}\\
        \text{части (1) не хватает элемента $q_{m_{n-1}}$. Давайте вычтем его справа и слева, получим:}
        \begin{equation}
            S-q_{m_{n-1}}\leq S_{k_{n-1}+m_{n-1}}
        \end{equation}
        \text{Объединив (7), (8) и (9), получим следующее неравенство:}
        \begin{equation}
            S-q_{m_{n-1}}\leq S_{k_{n-1}+m_{n-1}} <S_N\leq S
        \end{equation}
        \par{\text{Пусть мы перебрали элементов и перескочили S, тогда для N}\\\text{справедливо неравенство:}\\
        \begin{equation}
            \underbrace{k_n+m_{n-1}}_{\text{на предыд. шаге}}< N\leq\underbrace{k_n+m_n-1}_{\text{текущий шаг}}<\underbrace{k_n+m_n}_{\text{на след. шаге}}
        \end{equation}
        \text{Тогда $S_N$ точно меньше, чем сумма, полученная на следующем шаге (4), т.к.}\\
        \text{перебрали и добавили ещё один положительный элемент. Тогда: }
        \begin{equation}
            S_N<S_{k_n+m_{n-1}}
        \end{equation}
        \text{$k_n+m_n-1\leq N$, что соответсвует условию (5), но тогда:}
        \begin{equation}
            S\leq S_N
        \end{equation}
        \text{Но что такое $S_{k_n+m_{n-1}}$? Посмотрим на (3) и заметим, что если слева добавить $p_{k_n}$,}\\
        \text{то тогда получим нужную сумму. Но тогда и справа необходимо добавить $p_{k_n}$. Тогда:}
        \begin{equation}
            S_{k_n+m_{n-1}}\leq S+p_{k_n}
        \end{equation}
        \text{Объединив (12), (13), (14), получим следующую цепочку неравенств:}
        \begin{equation}
            S\leq S_N<S_{k_n+m_{n-1}}\leq S+p_{k_n}
        \end{equation}
        \text{Заметим, что (10) и (15) были получены из абсолютно аналогичных соображений.}\newpage
        \text{Теперь объединим наше знание:}
        \begin{equation}
            S-p_{k_n}<S\leq S_N<S+p_{k_n}
        \end{equation}
        \begin{equation}
            S-q_{m_{n-1}} < S_N \leq S <S+q_{m_{n-1}}
        \end{equation}
        \text{Тогда возьмём $\max\{p_{k_n} \ , \ q_{m_{n-1}}\}$, тогда из (16) и (17) для любого шага верно, что}
        \begin{equation}
            S-\max\{p_{k_n} \ , \ q_{m_{n-1}}\} < S_N  <S+\max\{p_{k_n} \ , \ q_{m_{n-1}}\}
        \end{equation}
        \text{А из (18) следует, что $|S_N-S|<\max\{p_{k_n} \ , \ q_{m_{n-1}}\}$. Т.к. ряд $\displaystyle\sum_{n=1}^{\infty}a_n$ сходится, то $a_n\rightarrow 0$}\\
        \text{по лемме 4 $\S1$, а отсюда следует, что $\max\{p_{k_n} \ , \ q_{m_{n-1}}\}\rightarrow0\Rightarrow\displaystyle\sum_{n=1}^{\infty}\tilde{a}_n=S$.}
        }
        \par{
        \text{Пусть $S=\pm\infty$. Тогда будем идти такими шагами:}
        \begin{enumerate}
            \item \par{
            \begin{enumerate}
                \item \par{$p_1+\ldots+p_{k_1-1}\leq1$}
                \item \par{$p_1+\ldots+p_{k_1-1}+p_{k_1}>1$}
            \end{enumerate}}
            \item \par{
            \begin{enumerate}
                \item \par{$p_1+\ldots+p_{k_1}-q_1+p_{k_1+1}+\ldots+p_{k_2-1}\leq2$}
                \item \par{$p_1+\ldots+p_{k_1}-q_1+p_{k_1+1}+\ldots+p_{k_2-1}+p_{k_2}>2$}
            \end{enumerate}
            }
        \end{enumerate}
        } 
        \ldots \\
        \text{Тогда на шаге $n$ получим следующее соотношение:}
        \begin{equation*}
            p_1+\ldots+p_{k_n-1}-q_1-\ldots-q_{n-1} \leq n
        \end{equation*}
        \begin{equation}
            \underbrace{p_1+\ldots+p_{k_n}-q_1-\ldots-q_{n-1}}_{S_{{k_n}+n-1}} > n
        \end{equation}
        \text{Тогда на шаге $n+1$ получим следующее соотношение:}
        \begin{equation*}
            p_1+\ldots+p_{k_{n+1}-1}-q_1-\ldots-q_{n} \leq n+1
        \end{equation*}
        \begin{equation*}
            p_1+\ldots+p_{k_{n+1}}-q_1-\ldots-q_{n} > n+1
        \end{equation*}
        \text{Пусть мы сейчас на шаге $n+1$ и уже выбрали N слагаемых. Тогда можно сказать, что}
        \begin{equation}
            \underbrace{k_n}_{+}+\underbrace{n}_{-}<N\leq \underbrace{k_{n+1}+n}_{\text{текущий шаг}}
        \end{equation}
        \text{Тогда из (20) следует, что $S_N$ уже точно больше $S_{{k_n}+n}$, т.к. $S_N$ был получен на шаге}\\
        \text{$n+1$ из $S_{{k_n}+n}$ из шага $n$ путём добавления неотрицательных слагаемых (по построению).}\\
        \begin{equation}
            S_{{k_n}+n}<S_N
        \end{equation}
        При этом из (19) следует, что
        \begin{equation}
            n-q_n<S_{k_n+n}
        \end{equation}
        \text{Объединив (21) и (22), получим, что $n-q_n<S_N$, при этом $n\rightarrow\infty$, а $q_n\rightarrow0$. Отсюда}\\
        \text{Следует, что $S_N$ при таком построении тоже стремится к бесконечности.}\\
        \text{При этом наше построение возможно, т.к. на каждом шаге мы берём не менее 1}\\
        \text{отрицательного и не менее 1 положительного членов. Т.о. покроем весь изначальный ряд.}\\
        \text{Для $-\infty$ нужно поменять p и q местами, т.е. набирать "больше" \ q, чем p.}
        \par{{\raggedleft ч.т.д.\par}}

        \section*{$\S$4. Функциональные ряды}
        \addcontentsline{toc}{section}{$\S$4. Функциональные ряды}
        \subsection*{1. Равномерная сходимость}
        \addcontentsline{toc}{subsection}{1. Равномерная сходимость}
        \par{\text{$\displaystyle\sum_{n=1}^{\infty}f_n(x), \ S_N(x)=\sum_{n=1}^{N}f_n(x)$. Ряд $\displaystyle\sum_{n=1}^{\infty}f_n(x)$ сходится $\Leftrightarrow \exists\displaystyle\lim_{N\rightarrow\infty}S_N(x)$.}\\}
        \par{\textbf{Def 1: }\text{Последовательность $\{S_N\}$ сходится к S равномерно на множестве I,}\\
        \text{если $\forall\epsilon>0 \ \exists N_0\in\mathbb{N}$, т.ч. $\forall n>N_0$ и $\forall x\in I$: $|S_N(x)-S(x)|<\epsilon$.}\\
        \text{Замечание: $N_0$ зависит от $\epsilon$, но не зависит от $x$. Обозначают $\displaystyle S_N(x){\rightrightarrows}_{N\rightarrow\infty}S(x)$.}
        }
        \par{\textbf{Def 2: } \text{Ряд $\displaystyle\sum_{n=1}^{\infty}f_n(x)$ сходится к $S(x)$ равномерно на мн-ве $I$, если $S_N(x)\rightrightarrows S(x)$ на I.}}
        \textbf{Примеры:}
        \begin{enumerate}
            \item \par{
            \text{$S_N(x)=x^N$, $x\in(0,1)$: $\displaystyle\lim_{N\rightarrow\infty}x^N=0$ $\forall x\in(0,1)$}\\
            \text{Пусть $\epsilon_0={1\over2}$ и $N_0$ - любое. Тогда $x^N={1\over2}$, $x=\sqrt[N]{1\over2}\xrightarrow[N\rightarrow\infty]{}1\in(0, 1)$.}\\
            \text{Т.е. $x^N<{1\over2}$ неверно $\forall x\in(0,1)$.}
            }
            \item \par{\text{$S_N(x)=x^N$, $x\in[0, q]$, где $q\in(0,1)$. Тогда $x^N\leq q^N<\epsilon \Rightarrow e^{N\ln{q}}<e^{\ln{\epsilon}}\Rightarrow$}\\
            \text{$\Rightarrow N>\frac{\ln{\epsilon}}{\ln{q}}$. Тогда возьмём $N_0=[\frac{\ln{\epsilon}}{\ln{q}}]$}
            }
            \item \par{
            \text{$\displaystyle\sum_{n=0}^{\infty}x^n={1\over{1-x}}, \ x\in(-1,1)$. $S_N(x)=\frac{x^{N+1}-1}{x-1}$ $\underbrace{[q_0, q_1]}_{\text{равномерно}} \subset \underbrace{(-1,1)}_{\text{неравномерно}}$.}
            }
        \end{enumerate}
        \par{}
        \vspace{0.8cm}
        \addcontentsline{toc}{subsubsection}{Теорема (Критерий Коши для равномерной сходимости)}
        \setcounter{equation}{2}
        \par{
        \textbf{Теорема (Критерий Коши для равномерной сходимости)}\\
        \text{(1) Последовательность $S_N$ сходится равномерно на множестве $I \Leftrightarrow \forall\epsilon>0 \ \exists N_0\in\mathbb{N}$, т.ч.}\\
        \text{$\forall N,M>N_0 \ \ \ \forall x\in I \ \ \ \ |S_N(x)-S_M(x)|<\epsilon$.}\\
        \text{(2) Ряд $\displaystyle\sum_{n=1}^{\infty}f_n(x)$ сходится равномерно на множестве $I \Leftrightarrow \forall\epsilon>0 \ \exists N_0\in\mathbb{N}$, т.ч.} \\
        \text{$\forall n>N$ $\forall p\in\mathbb{N}$ $\forall x\in I$: $|f_{n+1}(x)+f_{n+2}(x)+\ldots+f_{n+p}(x)|<\epsilon$.} \\ \\
        \textbf{Доказательство: } \textbf{[$\Rightarrow$] } \text{Пусть $S_N \rightrightarrows S(x)$ на I, т.е. $\forall\epsilon>0 \ \exists N_0\in\mathbb{N}$: $\forall N>N_0$ $\forall x\in I$}\\
        \text{$|S_N(x)-S(x)| < \epsilon$. Тогда $\forall N,M>N_0$ имеем: }\\
        \text{$|S_N(x)-S_M(x)|\leq|S_N(x)-S(x)|+|S_M(x)-S(x)|<2\epsilon$ $\forall x\in I$.}\\
        \\
        \textbf{[$\Leftarrow$] } \text{Пусть $\forall\epsilon>0 \ \exists N_0\in\mathbb{N}$: $\forall N,M>N_0 \ \ \ \forall x\in I$ выполнено следующее соотношение:}\\
        \begin{equation}
            |S_N(x)-S_M(x)|<{\epsilon\over2}
        \end{equation}
        \text{Фикс. $x\in I$, тогда $S_N(x)$ - числовая последовательность, удовлетворяющая}\\
        \text{критерию Коши для числовых последовательностей. По кр. Коши $\displaystyle\exists\lim_{N\rightarrow\infty}S_N(x)=S(x)$.}\\
        \text{Устремим $M\rightarrow\infty$. Тогда $S_M(x)\rightarrow S(x)$. При переходе к пределу в (3) получим, что}\\
        \text{$|S_N(x)-S(x)|\leq{\epsilon\over2}<\epsilon$ $\forall x\in I$.}
        \\ \\
        \text{Докажем (2): ряд $\displaystyle\sum_{n=1}^{\infty}f_n(x)$ сходится равномерно $\Leftrightarrow S_N(x)$ сходится равномерно.} \\
        \text{Применим первую часть доказательства при $M=N+p$: $\forall\epsilon>0 \ \exists N_0\in\mathbb{N}$, т.ч.}\\
        \text{$\forall n>N_0$: $|S_{N+p}-S_N|=|f_{n+1}(x)+f_{n+2}(x)+\ldots+f_{n+p}(x)|<\epsilon$.}
        }
        \par{{\raggedleft ч.т.д.\par}}

        \subsection*{2. Признаки равномерной сходимости}
        \addcontentsline{toc}{subsection}{2. Признаки равномерной сходимости}
        \par{\ }
        \addcontentsline{toc}{subsubsection}{Теорема (признак Вейерштрасса)}
        \par{\textbf{Теорема (признак Вейерштрасса)}\\
        \text{Пусть $f_n$ определена на множестве $I\subset\mathbb{R}$ и $|f_n(x)|\leq a_n$ $\forall x\in I$ и ряд $\displaystyle\sum_{n=1}^{\infty}a_n$ сходится.} \\
        \text{Тогда ряд $\displaystyle\sum_{n=1}^{\infty}f_n(x)$ сходится абсолютно $\forall x\in I$ и равномерно на $I$.}\\
        \textbf{Доказательство: } \text{Ряд $\displaystyle\sum_{n=1}^{\infty}a_n$ сходится и $|f_n(x)|\leq a_n$ $\forall x\in I$, тогда по признаку}\\
        \text{сравнения ряд $\displaystyle\sum_{n=1}^{\infty}|f_n(x)|$ сходится, тогда ряд $\displaystyle\sum_{n=1}^{\infty}f_n(x)$ сходится абсолютно.}\\
        \text{Зафиксируем $\epsilon>0$ и выберем $N\in\mathbb{N}$, т.ч. $a_{n+1}+\ldots+a_{n+p}<\epsilon$ $\forall n>N, p\in\mathbb{N}$. Тогда}\\
        \text{для указанных $n$ и $p$ $|f_{n+1}(x)+\ldots+f_{n+p}(x)|\leq|f_{n+1}(x)|+\ldots+|f_{n+p}(x)| \leq$}\\
        \text{$\leq a_{n+1}+\ldots+a_{n+p}<\epsilon$ $\forall x\in I$. Тогда по кр. Коши $\displaystyle\sum_{n=1}^{\infty}f_n(x)$ сходится равномерно на $I$.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.8cm}
        \par{
        \textbf{Пример: } \text{$\displaystyle\sum_{n=1}^{\infty}\frac{(-1)^n}{n}x^n, x\in[0,1]$. $\frac{x^n}{n}$ монотонно убывает к 0, тогда по признаку}\\
        \text{Лейбница ряд сходится $\forall x\in I$. Пусть $x\in[0,q]$, где $q<1$, тогда $|\frac{(-1)^n}{n}x^n|\leq q^n$ и $q<1$.}\\
        \text{Ряд $\displaystyle\sum_{n=1}^{\infty}q^n$ сходится, тогда по признаку Вейерштрасса наш ряд сходится равномерно.}
        }
        \vspace{0.5cm}
        \par{\textbf{Следствие 1. Признак Коши}\\
        \text{Пусть $f_n$ определена на $I\subset\mathbb{R}$ и, начиная с некоторого номера, 
        $\sqrt[n]{|f_n(x)|}\leq q_0<1$ $\forall x\in I$,}\\
        \text{тогда ряд $\displaystyle\sum_{n=1}^{\infty}f_n(x)$ сходится равномерно на $I$.}\\
        \textbf{Доказательство: } \text{$\displaystyle|f_n(x)|\leq q_0^n$. $\displaystyle\sum_{n=1}^{\infty}q_0^n$ сходится $\Rightarrow$ по теореме выше ряд $\displaystyle\sum_{n=1}^{\infty}f_n(x)$}\\
        \text{сходится абсолютно $\forall x\in I$ и равномерно на $I$.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.8cm}
        \setcounter{equation}{0}
        \subsection*{3. Свойства суммы равномерно сходящегося ряда}
        \addcontentsline{toc}{subsection}{3. Свойства суммы равномерно сходящегося ряда}
        \par{\ }
        \par{\textbf{Теорема 1. } \text{Пусть последовательность $S_n(x)$ равномерно сходится на $I\subset\mathbb{R}$.}\\
        \text{Если $S_n(x)$ непрерывна на $I$, то и $S(x)$ непрерывна на $I$.}\\
        \textbf{Доказательство: } \text{фикс. $\epsilon>0$ и выберем $N\in\mathbb{N}$ так, что $\forall n>N$ и $\forall x\in I$:}\\
        \begin{equation}
            |S_n(x)-S(x)|<\epsilon
        \end{equation}
        \text{Пусть $x_0$ - любая точка, $n_0>N$ - фиксировано. Т.к. $S_n$ непрерывна в точке $x_0$, то}\\
        \text{$\exists\delta>0$, что $\forall x\in B_{\delta}(x_0)\cap I$ верно:}
        \begin{equation}
            |S_{n_0}(x)-S_{n_0}(x_0)|<\epsilon
        \end{equation}
        \newpage
        \text{Тогда для таких x: $|S(x)-S(x_0)|=|S(x)-S_{n_0}(x)+S_{n_0}(x)-S_{n_0}(x_0)+S_{n_0}(x_0)-S(x_0)|\leq$}\\
        \text{$\leq\underbrace{|S(x)-S_{n_0}(x)|}_{<\epsilon\text{ из (1)}}+\underbrace{|S_{n_0}(x)-S_{n_0}(x_0)|}_{<\epsilon\text{ из (2)}}+\underbrace{|S_{n_0}(x_0)-S(x_0)|}_{<\epsilon\text{ из (1)}}<3\epsilon$}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.6cm}
        \par{\textbf{Следствие 1. } \text{Пусть ряд $\displaystyle\sum_{n=1}^{\infty}f_n(x)$ сходится равномерно на $I\subset\mathbb{R}$, а его члены}\\
        \text{непрерывны на $I$. Тогда $S(x)=\displaystyle\sum_{n=1}^{\infty}f_n(x)$ непрерывна на $I$.}\\
        \textbf{Доказательство: } \text{Т.к. $f_n$ непрерывна, то и $S_N$ непрерывна на $I$. Т.к. $S_N(x)\rightrightarrows S(x)$,}\\
        \text{то по теореме выше $S(x)$ непрерывна на $I$.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.8cm}
        \setcounter{equation}{0}
        \par{\textbf{Теорема 2 (интегрируемость)} \\
        \text{Пусть последовательность $S_n(x)$ равномерно сходится к $S(x)$ на $[a, b]\subset\mathbb{R}$. Если}\\
        \text{$S_n(x)$ интегрируема на $[a,b]$, то и $S(x)$ интегрируема на $[a,b]$.}\\
        \textbf{Доказательство: } \text{фикс. $\epsilon>0$ и выберем $N\in\mathbb{N}$, что $\forall n>N$ и $\forall x\in [a,b]$ верно}\\
        \text{$|S_n(x)-S(x)|<\epsilon$. Рассмотрим какое-нибудь разбиение $a=x_0<x_1<\ldots<x_{N+1}=b$.}\\
        \text{Пусть $\displaystyle\sum^* f(\xi_i)\Delta x_i$ - верхняя сумма Дарбу, а$\displaystyle\sum_* f(\xi_i)\Delta x_i$ - нижняя сумма Дарбу.}\\
        \begin{equation}
            \sum_{i=0}^{N+1}\epsilon\Delta x_i=\epsilon\sum_{i=0}^{N+1}\Delta x_i=\epsilon(x_1-x_0+x_2-x_1+\ldots+x_{N+1})=\epsilon(x_{N+1}-x_0)=\epsilon(b-a)
        \end{equation}
        \text{Тогда из $S_n(x)-\epsilon<S(x)<S_n(x)+\epsilon$ домножением на $\Delta x_i$ и суммированием получаем,}\\
        \begin{equation*}
            \sum_*S_n(\xi_i)\Delta x_i-\underbrace{\epsilon(b-a)}_{\text{из (1)}}\leq\sum_*S(\xi_i)\Delta x_i\leq\sum^*S(\xi_i)\Delta x_i\leq\sum^*S_n(\xi_i)\Delta x_i+\underbrace{\epsilon(b-a)}_{\text{из (1)}}
        \end{equation*}
        \text{Тогда из этой формулы следует, что}
        \begin{equation}
            \sum^*S(\xi_i)\Delta x_i-\sum_*S(\xi_i)\Delta x_i\leq\sum^*S_n(\xi_i)\Delta x_i-\sum_*S_n(\xi_i)\Delta x_i+2\epsilon(b-a)
        \end{equation}
        \text{Пусть $n>N$ фиксировано, тогда $\exists\delta>0$, что из $\lambda<\delta\Rightarrow$ (3).}
        \begin{equation}
            \sum^*S_n(\xi_i)\Delta x_i-\sum_*S_n(\xi_i)\Delta x_i <\epsilon
        \end{equation}
        \text{Подставив (3) в правую часть (2) получим следующее неравенство:}
        \begin{equation*}
            \sum^*S(\xi_i)\Delta x_i-\sum_*S(\xi_i)\Delta x_i<\epsilon+2\epsilon(b-a)
        \end{equation*}
        \text{Т.е. разница между верхней и нижней суммами Дарбу меньше наперёд заданного числа.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.6cm}
        \par{
        \textbf{Следствие из теоремы 2: }\text{Если $\displaystyle\sum_{n=1}^{\infty}f_n(x)$ равномерно сходится к $S(x)$ на $[a,b]$, а}\\
        \text{$f_n$ интегрируема на $[a,b]$, то и $S$ интегрируема на $[a,b]$.}
        }

        \subsection*{4. Предельный переход под знаком интеграла. Интегрирование рядов}
        \addcontentsline{toc}{subsection}{4. Предельный переход под знаком интеграла. Интегрирование рядов}
        \par{\ }
        \par{\textbf{Теорема 3. }\text{Пусть $S_n(x)$ интегрируема на $[a,b]$ и $S_n(x)\rightrightarrows S(x)$ на $[a,b]$. Тогда}\\
        \text{последовательность $\displaystyle\{\int_{x_0}^{x}S_n(t)dt\}$ сходится к $\displaystyle\int_{x_0}^{x}S(t)dt$ равномерно по $x\in[a,b]$ при}\\
        \text{$\forall$ фиксированном $x_0\in[a,b]$. Т.е. $\displaystyle\lim_{n\rightarrow\infty}\int_{x_0}^{x}S_n(t)dt=\int_{x_0}^{x}(\lim_{n\rightarrow\infty}S_n(t))dt$.}\\
        \textbf{Доказательство: }\text{т.к. $S_n$ интегрируема на $[a,b]$ и $S_n(x)\rightrightarrows S(x)$ на $[a,b]$, то по теореме 2}\\
        \text{$S$ интегрируема на $[a,b]$. Зафиксируем $\epsilon>0$, тогда из $S_n(x)\rightrightarrows S(x)\Rightarrow\exists N\in\mathbb{N}$, что}\\
        \text{$\forall n>N$ и $\forall t\in[a,b]$ выполнено: $|S_n(t)-S(t)|<\epsilon$. Тогда $\forall n>N$ и $\forall x,x_0\in[a,b]$ выполнено:}\\
        \text{$\displaystyle|\int_{x_0}^{x}S_n(t)dt-\int_{x_0}^{x}S(t)dt|=|\int_{x_0}^{x}(S_n(t)-S(t))dt|\leq|\int_{x_0}^{x}|S_n(t)-S(t)|dt|\leq|\int_{x_0}^{x}\epsilon \ dt|<$}\\
        \text{$\displaystyle<\epsilon|\int_{x_0}^{x}dt|=\epsilon|x-x_0|\leq\epsilon(b-a)$. Т.е. $\displaystyle\int_{x_0}^{x}S_n(t)dt\rightarrow\int_{x_0}^{x}S(t)dt$ равномерно на $x\in[a,b]$.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.6cm}
        \par{\textbf{Def 3: } \text{Пусть $f_n$ интегрируема на $[a,b]$ и ряд $\displaystyle\sum_{n=1}^{\infty}f_n(x)=S(x)$ $\forall x\in[a,b]$.}\\
        \text{Будем говорить, что $\displaystyle\sum_{n=1}^{\infty}f_n(x)$ почленно интегрируем на $[a,b]$, если его сумма $S(x)$}\\
        \text{интегрируема на $[a,b]$ и $\displaystyle\sum_{n=1}^{\infty}\int_{x_0}^{x}f_n(t)dt=\int_{x_0}^{x}S(t)dt$ для $\forall x,x_0\in[a,b]$.}
        }
        \vspace{0.8cm}
        \par{
        \textbf{Следствие: }\text{Пусть члены ряда $\displaystyle\sum_{n=1}^{\infty}f_n(x)$ интегрируемы на $[a,b]$ и $\displaystyle\sum_{n=1}^{\infty}f_n(x)\rightrightarrows S(x)$,}\\
        \text{тогда ряд $\displaystyle\sum_{n=1}^{\infty}f_n(x)$ почленно интегрируем на $[a,b]$, причём $\displaystyle\sum_{n=1}^{\infty}\int_{x_0}^{x}f_n(t)dt$ сходится}\\
        \text{к $\displaystyle\int_{x_0}^{x}S(t)dt$ равномерно по $x\in[a,b]$ для $\forall$ фиксированного $x_0\in[a,b].$}\\
        \textbf{Доказательство: }\text{т.к. $f_n$ интегрируема на $[a,b]$, то и $S_N$ интегрируема на $[a,b]$. Т.к. }\\
        \text{$S_N\rightrightarrows S$ на $[a,b]$, то $S$ интегрируема на $[a,b]$ и по теореме 3 $\displaystyle\int_{x_0}^{x}S_n(t)dt\rightrightarrows\int_{x_0}^{x}S(t)dt$}\\
        \text{по $x\in[a,b]$ $\forall x_0\in[a,b]$.}}
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.6cm}
        \par{\textbf{Пример: } \text{$S_n=nx^n(1-x^n)$, $x\in[0,1]$.}
        \begin{equation*}
            \int^1_0S_n(t)dt=\int^1_0nt^n(1-t^n)dt=n(\frac{x^{n+1}}{n+1}-\frac{x^{2n+1}}{2n+1})\Bigg|^1_0=\frac{n^2}{(n+1)(2n+1)}\rightarrow{1\over2}
        \end{equation*}
        \text{Но при этом $S_n(x)\rightarrow0$ при $\forall x\in[0,1]$. Нетрудно заметить, что нарушена равномер-}\\
        \text{ная сходимость.}
        }

        \subsection*{5. Дифференцирование функциональных рядов}
        \addcontentsline{toc}{subsection}{5. Дифференцирование функциональных рядов}
        \par{\ }
        \par{
        \textbf{Теорема 4 (дифференцирование функциональных рядов)}\\
        \text{Пусть последовательность $S_n(x)$ удовлетворяет следующим трём условиям:}
        \begin{enumerate}
            \item \par{$S'_n(x)$ $\exists$ и непрерывна на $[a,b]$;}
            \item \par{$S'_n(x)\rightrightarrows\sigma(x)$ на $[a,b]$;}
            \item \par{Для некоторого $x_0\in[a,b]$ $\displaystyle\exists\lim_{n\rightarrow\infty}S_n(x_0)=l$.}
        \end{enumerate}
        \text{Тогда $\sigma$ непрерывна на $[a,b]$ и $\displaystyle\exists\lim_{n\rightarrow\infty}S_n(x)=\int^{x}_{x_0}\sigma(t)dt+l$, сходимость равномерно по}\\
        \text{\ }\\
        \text{$x\in[a,b].$ Т.е. предел $S(x)$ дифференцируем и $S'(x)=\sigma(x).$}\\
        \text{$S'=\sigma\Leftrightarrow(\lim S_n(x))'=\lim S'_n(x)$}\\
        \textbf{Доказательство: }\text{т.к. $S_n'$ непрерывна и $S'_n(x)\rightrightarrows\sigma(x)$ на $[a,b]$, то $\sigma$ непрерывна на $[a,b]$,}\\
        \text{а потому интегрируема на $[a,b]$. Т.к. $S'_n(x)\rightrightarrows\sigma(x)$ на $[a,b]$, то}\\
        \begin{equation}
            \int_{x_0}^{x}S'_n(t)dt\rightrightarrows\int_{x_0}^{x}\sigma(t)dt
        \end{equation}
        \text{по $x\in[a,b]$ и с $x_0$ из пункта 3 данной теоремы, а для него верно:}\\
        \begin{equation}
            \exists\lim_{n\rightarrow\infty}S_n(x_0)=l
        \end{equation}
        \text{Применяя формулу Ньютона-Лейбница к $S'_n(x)$ получим:}
        \begin{equation}
            S_n(x)=S_n(x_0)+\int_{x_0}^{x}S'_n(t)dt
        \end{equation}
        \text{Но в силу (1) и (2), конструкция (3) стремится равномерно по $x\in[a,b]$ к (4)}
        \begin{equation}
            S_n(x)\rightrightarrows l+\int_{x_0}^{x}\sigma(t)dt
        \end{equation}
        \text{Т.к. $\sigma$ непрерывна, то $S'(x)=\sigma(x)$ в силу теоремы о дифференцировании определённого}\\
        \text{интеграла по переменному верхнему пределу.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.3cm}
        \par{\textbf{Следствие: } \text{Пусть ряд $\displaystyle\sum_{n=1}^{\infty}f_n(x)$ удовлетворяет следующим трём условиям:}
        \begin{enumerate}
            \item \par{\text{$f_n(x)$ непрерывно дифференцируема на $[a,b]$;}}
            \item \par{\text{$\displaystyle\sum_{n=1}^{\infty}f_n(x)$ сходится равномерно на $[a,b]$;}}
            \item \par{\text{$\exists x_0\in[a,b]$ такой, что $\displaystyle\sum_{n=1}^{\infty}f_n(x_0)$ сходится.}}
        \end{enumerate}
        \text{Тогда ряд $\displaystyle\sum_{n=1}^{\infty}f'_n(x)$ сходится равномерно на $[a,b]$ и $\displaystyle\sum_{n=1}^{\infty}f_n(x)=\sum_{n=1}^{\infty}f_n(x_0)+\int^{x}_{x_0}\bigg(\sum_{n=1}^{\infty}f'_n(t)\bigg)dt$,}\\
        \text{т.е. сумма ряда дифференцируема и $\displaystyle\bigg(\sum_{n=1}^{\infty}f_n(x)\bigg)' = \sum_{n=1}^{\infty}f'_n(x)$.}
        }
        \newpage
        \par{\textbf{Пример: } \text{$\displaystyle\sum_{n=1}^{\infty}\frac{\sin({n^2x})}{n^2}$, $x\in\mathbb{R}$. $\displaystyle\Bigg|\frac{\sin({n^2x})}{n^2}\Bigg|\leq\frac{1}{n^2}\Rightarrow$ ряд сходится, но при этом ряд}\\
        \text{из производных $\displaystyle\sum_{n=1}^{\infty}\frac{2\cos({n^2x})}{n}$ расходится.}
        }

        \section*{$\S$5. Степенные ряды}

        \subsection*{1. Определение. Радиус сходимости}
        \addcontentsline{toc}{subsection}{1. Определение. Радиус сходимости}
        \par{\text{$\displaystyle\sum_{n=0}^{\infty}a_n(x-x_0)^n$ - степенной ряд, $a_n$ - коэффициенты ряда.}}
        \vspace{0.3cm}
        \addcontentsline{toc}{subsubsection}{Теорема (Абель)}
        \par{\textbf{Теорема (Абель)}\\
        \text{Пусть ряд $\displaystyle\sum_{n=0}^{\infty}a_n(x-x_0)^n$ сходится при $x=x_1$, тогда он сходится абсолютно}\\
        \text{$\forall x\in(x_0-|x_0-x_1|, \ x_0+|x_0-x_1|)$ и равномерно на $\forall$ отрезке из этого интервала.}\\
        \textbf{Доказательство: }\text{если $\displaystyle\sum_{n=0}^{\infty}a_n(x-x_0)^n$ сходится, то $a_n(x_1-x_0)^n\rightarrow0$, тогда $|a_n(x_1-x_0)^n|\leq c$}\\
        \text{$\forall n\in\mathbb{N}$. Отсюда получаем, что}
        \begin{equation}
            |a_n|\leq\frac{c}{|x_1-x_0|^n}
        \end{equation}
        \text{Рассмотрим отрезок $[a,b]\subset(x_0-|x_0-x_1|, \ x_0+|x_0-x_1|)$ и $x\in[a,b]$, тогда}
        \begin{equation}
            |x-x_0|<|x_1-x_0| \text{ и при этом из (1) }\Rightarrow |a_n(x-x_0)^n|\leq c\bigg|\frac{x-x_0}{x_1-x_0}\bigg|^n 
        \end{equation}
        \text{Пусть a - наиболее удалённая точка на $[a,b]$ от $x_0$, тогда из $\displaystyle(2)\Rightarrow |a_n(x-x_0)^n|\leq c\bigg|\frac{a-x_0}{x_1-x_0}\bigg|^n$.}\\
        \text{Т.к. $|a-x_0|<|x_1-x_0|$, то ряд $\displaystyle\sum_{n=0}^{\infty}c\bigg|\frac{a-x_0}{x_1-x_0}\bigg|^n$ сходится по признаку Коши, то наш ряд}\\
        \text{$\displaystyle\sum_{n=0}^{\infty}a_n(x-x_0)^n$ сходится по признаку Вейерштрасса абсолютно $\forall x\in[a,b]$ и равномерно}\\
        \text{на $[a,b]$. Т.к. $[a,b]\subset(x_0-|x_0-x_1|, \ x_0+|x_0-x_1|)$ выбирался произвольно, то ряд}\\
        \text{$\displaystyle\sum_{n=0}^{\infty}a_n(x-x_0)^n$ сходится абсолютно $\forall x$ из этого интервала.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \par{\textbf{Обозначим за R} \text{радиус сходимости, а за область сходимости - интервал $<x_0-R, \ x_0+R>$.}\\
        \text{При этом < - это [ или (. Т.е. про концы такого отрезка теорема Абеля ничего сказать не может.}
        }
        \par{
        \textbf{Примеры:}}
        \begin{enumerate}
            \item \par{\text{$\displaystyle\sum_{n=1}^{\infty}(nx)^n$}. R=0, т.к. $(nx)^n\nrightarrow0$}
            \item \par{\text{$\displaystyle\sum_{n=1}^{\infty}\frac{x^n}{n!}=e^x \ \forall x\in\mathbb{R}\Rightarrow R=\infty$.}}
            \newpage
            \item \par{\text{$\displaystyle\sum_{n=1}^{\infty}x^n={1\over{1-x}}$ $\forall x\in(-1,1).$ Отсюда R=1.}}
        \end{enumerate}
        \subsection*{2. Формулы для R}
        \addcontentsline{toc}{subsection}{2. Формулы для R}
        \par{\ }
        \addcontentsline{toc}{subsubsection}{Теорема (формула для радиуса сходимости)}
        \par{\textbf{Теорема (формула для радиуса сходимости)}\\
        \text{Пусть R - радиус сходимости ряда $\displaystyle\sum_{n=0}^{\infty}a_n(x-x_0)^n$. Тогда $\displaystyle{1\over{R}}=\overline{\lim_{n\rightarrow\infty}}\sqrt[n]{a_n}$.}\\
        \textbf{Доказательство: }\text{Выразим обобщённый признак Коши через радиус сходимости:}
        \begin{equation}
            \displaystyle\overline{\lim_{n\rightarrow\infty}}\sqrt[n]{a_n(x-x_0)^n}=|x-x_0|\overline{\lim_{n\rightarrow\infty}}\sqrt[n]{a_n}=|x-x_0|{1\over{R}}
        \end{equation}
        \begin{itemize}
            \item \par{\text{Пусть $|x-x_0|<R$, тогда $(3)<1\Rightarrow$ ряд сходится по обобщённому пр-ку Коши;}}
            \item \par{\text{Пусть $|x-x_0|>R$, тогда $(3)>1\Rightarrow$ ряд расходится по обобщённому пр-ку Коши;}}
        \end{itemize}
        \text{Тогда формула для $\displaystyle1\over{R}$ удовлетворяет условиям.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.3cm}
        \par{\text{Также $\displaystyle{1\over{R}}=\lim_{n\rightarrow\infty}\bigg|\frac{a_{n+1}}{a_n}\bigg|$. Упражнение: доказать самостоятельно.}\\
        \textbf{Примеры:}}
        \begin{enumerate}
            \item \par{\text{$\displaystyle\sum_{n=0}^{\infty}(nx)^n$. $a_n=n^n$, $\sqrt[n]{a_n}=n\rightarrow\infty$ $\Rightarrow$ $R=0$}}
            \item \par{\text{$\displaystyle\sum_{n=0}^{\infty}\frac{x^n}{n!}$, $\displaystyle a_n={1\over{n!}}$. $\displaystyle \frac{a_{n+1}}{a_n}={1\over{n+1}}\rightarrow0\Rightarrow R\rightarrow+\infty$}}
            \item \par{\text{$\displaystyle\sum_{n=0}^{\infty}x^n$, $a_n=1\Rightarrow R=1$}}
        \end{enumerate}
        \par{\text{Концы, однако, придётся рассматривать отдельно. Формулы для радиуса не дают}\\
        \text{никакой информации о концах.}}
        \par{\text{Например, ряд $\displaystyle \sum_{n=0}^{\infty}\frac{(x-x_0)^n}{n}$, $\displaystyle a_n={1\over n}\Rightarrow R=1$. Но при $x-x_0=1$ - расходится, а при}\\
        \text{$x-x_0=-1$ - сходится.}}

        \subsection*{3. Равномерная сходимость в окрестности концов интервала сходимости}
        \addcontentsline{toc}{subsection}{3. Равномерная сходимость в окрестности концов интервала сходимости}
        \par{\ }
        \setcounter{equation}{0}
        \par{\textbf{Теорема. } \text{Пусть ряд $\displaystyle\sum_{n=0}^{\infty}c_n(x-x_0)^n$ сходится в точке $x_0+R$ (соотв. $x_0-R$), тогда он}\\
        \text{сходится равномерно на [$x_0$, $x_0+R$] (соотв. на [$x_0-R$, $x_0$]). Вместо $x_0$ можно выбрать}\\
        \text{$\forall x_1\in(x_0-R, x_0+R)$.}\\
        \textbf{Доказательство: }\text{рассмотрим $\displaystyle\sum_{q=1}^{p}c_{n+q}(x-x_0)^{n+q}=\sum_{q=1}^{p}(c_{n+q}R^{n+q})\bigg(\frac{x-x_0}{R}\bigg)^{n+q}$ - для}\newpage
        \text{подсказки см. критерий Коши.}\\
        \text{Применим преобразование Абеля: $\displaystyle\tilde{B}_p=\sum_{q=1}^{p}c_{n+q}R^{n+q}$, тогда получим:}
        \begin{equation*}
            \sum_{q=1}^{p}c_{n+q}(x-x_0)^{n+q}=\bigg(\frac{x-x_0}{R}\bigg)^{n+p}\tilde{B}_p+\sum_{q=1}^{p-1}\Bigg(\bigg(\frac{x-x_0}{R}\bigg)^{n+q}-\bigg(\frac{x-x_0}{R}\bigg)^{n+q+1}\Bigg)\tilde{B}_q\Rightarrow
        \end{equation*}
        \begin{equation*}
            \Rightarrow\Bigg|\sum_{q=1}^{p}c_{n+q}(x-x_0)^{n+q}\Bigg|=\Bigg|\bigg(\frac{x-x_0}{R}\bigg)^{n+p}\tilde{B}_p+\sum_{q=1}^{p-1}\Bigg(\bigg(\frac{x-x_0}{R}\bigg)^{n+q}-\bigg(\frac{x-x_0}{R}\bigg)^{n+q+1}\Bigg)\tilde{B}_q\Bigg|\leq
        \end{equation*}
        
        \begin{equation}
            \leq\Bigg|\bigg(\frac{x-x_0}{R}\bigg)^{n+p}\tilde{B}_p\Bigg|+\Bigg|\sum_{q=1}^{p-1}\Bigg(\bigg(\frac{x-x_0}{R}\bigg)^{n+q}-\bigg(\frac{x-x_0}{R}\bigg)^{n+q+1}\Bigg)\tilde{B}_q\Bigg|
        \end{equation}
        \text{Т.к. $x\in[x_0, x_0+R]$, то $x-x_0\geq 0$, поэтому $\displaystyle\Bigg|\bigg(\frac{x-x_0}{R}\bigg)^n\Bigg|=\bigg(\frac{x-x_0}{R}\bigg)^n$, но ещё $x-x_0\leq R \Rightarrow$}\\
        \text{$\Rightarrow\displaystyle\bigg(\frac{x-x_0}{R}\bigg)^{n+q}-\bigg(\frac{x-x_0}{R}\bigg)^{n+q+1}\geq0$, поэтому у (1) частично можно убрать модули:}
        \begin{equation}
            (1)=\bigg(\frac{x-x_0}{R}\bigg)^{n+p}|\tilde{B}_p|+\sum_{q=1}^{p-1}\Bigg(\bigg(\frac{x-x_0}{R}\bigg)^{n+q}-\bigg(\frac{x-x_0}{R}\bigg)^{n+q+1}\Bigg)|\tilde{B}_q|
        \end{equation}
        \text{Т.к. ряд по условию сходится в точке $x=x_0+R$, то отсюда следует, что ряд $\displaystyle\sum_{n=0}^{\infty}c_nR^n$}\\
        \text{тоже сходится. Поэтому имеем право применить для него критерий Коши:}\\
        \text{$\forall\epsilon>0$ $\exists N\in\mathbb{N}:$ $\forall n>N$ и $\forall p\in\mathbb{N}$ верно, что $|c_{n+1}R^{n+1}+\ldots+c_{n+p}R^{n+p}|=|\tilde{B}_p|<\epsilon$ $\Rightarrow$}
        \begin{equation}
            \Rightarrow (2)<\epsilon\Bigg(\bigg(\frac{x-x_0}{R}\bigg)^{n+p}+\sum_{q=1}^{p-1}\Bigg(\bigg(\frac{x-x_0}{R}\bigg)^{n+q}-\bigg(\frac{x-x_0}{R}\bigg)^{n+q+1}\Bigg)
        \end{equation}
        \text{Заметим, что степени $n+q$ и $n+q+1$ чередуются, а потому сокращаются между собой,}\\
        \text{а остаются лишь степени $n+1$ и $n+p$ $\Rightarrow$}
        \begin{equation}
            (3)=\epsilon\Bigg(\bigg(\frac{x-x_0}{R}\bigg)^{n+p}+\bigg(\frac{x-x_0}{R}\bigg)^{n+1}-\bigg(\frac{x-x_0}{R}\bigg)^{n+p}\Bigg)=\epsilon\bigg(\frac{x-x_0}{R}\bigg)^{n+1}\leq\epsilon
        \end{equation}
        \text{т.к. $x\in[x_0, x_0+R]$, то $\displaystyle x-x_0\leq R \Rightarrow\bigg(\frac{x-x_0}{R}\bigg)^{n+1}\leq1$.}\\
        \text{В силу того, что $n$ не зависит от $x$ по построению, мы имеем право применить критерий}\\
        \text{Коши и сказать, что из (4) ряд $\displaystyle\sum_{n=0}^{\infty}c_n(x-x_0)^n$ сходится равномерно на $[x_0, x_0+R]$.}\\
        \text{Случай $x_1\in[x_0-R, x_0]$ сводится к предыдущему заменой $y=\mu x_0-x\in[x_0, x_0+R]$.}\\
        \text{Рассмотрим $[x_1, x_0+R]=\underbrace{[x_1,x_0]}_{\text{по Абелю}}\cup\underbrace{[x_0,x_0+R]}_{\text{по теореме выше}} \Rightarrow$ на отрезке $[x_1, x_0+R]$ ряд сходится.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.3cm}
        \par{\textbf{Следствие 1: }\text{Если степенной ряд сходится на замкнутом отрезке, то на замкнутом}\\
        \text{отрезке ряд сходится равномерно.}}
        \par{\textbf{Следствие 2: }\text{Сумма степенного ряда непрерывна в любой точке его промежутка}\\\text{сходимости.}\newpage
        \textbf{Доказательство: }\text{Пусть $x\in[a,b]\subset(x_0-R,x_0+R)$. По теореме Абеля степенной ряд}\\
        \text{сходится равномерно на [a,b]. Т.к. $c_n(x-x_0)^n$ непрерывны, то и сумма ряда непрерывна}\\
        \text{на [a,b] $\Rightarrow$ непрерывна в точке $x$. Сейчас осталось рассмотреть его границы: $x=x_0+R$,}\\
        \text{тогда мы имеем дело с рядом $\displaystyle\sum_{n=0}^{\infty}c_nR^n$: пусть он сходится, тогда функциональный ряд}\\
        \text{сходится равномерно на $[x_0, x_0+R]$ по теореме выше. А ещё члены ряда непрерывны,}\\
        \text{тогда сумма ряда непрерывна на $[x_0, x_0+R]$. Случай $x=x_0-R$ аналогичен.}
        }
        \subsection*{4. Интегрирование и дифференцирование степенных рядов}
        \addcontentsline{toc}{subsection}{4. Интегрирование и дифференцирование степенных рядов}
        \par{\ }
        \par{\textbf{Теорема 1. } \text{Пусть ряд $\displaystyle\sum_{n=0}^{\infty}a_n(x-x_0)^n$ сходится на [a,b], тогда он почленно}\\
        \text{интегрируем на нём, т.е. ряд $\displaystyle \sum_{n=0}^{\infty}a_n\int_{x_1}^{x}(t-x_0)^ndt$ сходится и его сумма равна}\\
        \text{$\displaystyle\int_{x_1}^{x}\bigg(\sum_{n=0}^{\infty}a_n(t-x_0)^n\bigg)dt$ $\forall x,x_1\in[a,b]$.}\\
        \textbf{Доказательство: }\text{члены ряда интегрируемы, степенной ряд сходится равномерно на }\\
        \text{любом отрезке $[a,b]\subset$ $<x_0-R, x_0+R>$ $\Rightarrow$ ряд почленно интегрируем.}
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.7cm}
        \setcounter{equation}{0}
        \par{\textbf{Теорема 2. } \text{пусть R - радиус сходимости ряда $\displaystyle\sum_{n=0}^{\infty}a_n(x-x_0)^n$ и $R\not=0$. Тогда}\\
        \text{ряд $\displaystyle\sum_{n=1}^{\infty}a_n\cdot n(x-x_0)^{n-1}$ сходится на $(x_0-R, x_0+R)$ к $\bigg(\displaystyle\sum_{n=1}^{\infty}a_n(x-x_0)^{n}\bigg)'$. Радиус}\\ \text{сходимости при этом тоже равен R.}\\
        \textbf{Доказательство: }\text{1) члены ряда дифференцируемы;}\\
        \text{2) пусть $[a,b]\subset(x_0-R,x_0+R)$. Выберем такое $r\in(0,R)$, что ряд $\displaystyle\sum_{n=0}^{\infty}a_nr^n$ сходится и}\\
        \text{$[a,b]\subset(x_0-r,x_0+r)$.}\\
        \begin{center}
            \includegraphics[width=0.4\linewidth]{image.png}
        \end{center}
        \text{Так как ряд сходится, то $|a_nr^n|\leq c$ $\forall n\in\mathbb{N}$. Обозначим $r_0=\max|x-x_0|<r$. Обозначим}\\
        \text{$\displaystyle q={r_0\over{r}}<1$ по введённому $r_0$. Тогда верно, что}
        \begin{equation}
            \bigg|\frac{x-x_0}{r}\bigg|\leq \frac{r_0}{r}=q<1
        \end{equation}
        \text{Тогда верна следующая цепочка неравенств:}
        \begin{equation}
            |f_n(x)|=|n\cdot a_n(x-x_0)^{n-1}|=n\cdot|a_n|\cdot|x-x_0|^{n-1}\leq n\frac{c}{r}\bigg|\frac{x-x_0}{r}\bigg|^{n-1}\leq n\frac{c}{r}q^{n-1}=n\frac{c}{rq}q^n
        \end{equation}
        \text{Т.к. $q<1$ из (1), тогда ряд $\displaystyle\frac{c}{rq}\sum_{n=1}^{\infty}nq^n$ сходится, то по признаку Вейерштрасса ряд из}\newpage
        \text{производных $\displaystyle\sum_{n=1}^{\infty}n\cdot a_n(x-x_0)^{n-1}$ сходится равномерно на [a,b].}\\
        \text{Напоминание: признак Вейерштрасса гласит о том, что если $|f_n(x)|\leq b_n$ $\forall x\in I$ и}\\
        \text{ряд $\displaystyle\sum_{n=1}^{\infty}c_n$ сходится, то ряд $\displaystyle\sum_{n=1}^{\infty}f_n(x)$ сходится абсолютно $\forall x\in I$ и равномерно на $I$.}\\
        \text{В нашем случае $I=[a,b]$, $f_n(x)=n\cdot a_n(x-x_0)^{n-1}$, $\displaystyle b_n=\frac{c}{rq}nq^n$ и $\displaystyle\frac{c}{rq}=const$. При этом}\\
        \text{$|f_n(x)|\leq b_n$ из (2). Сходимость ряда из $b_n$ мы доказали. Тогда ряд из производных}\\
        \text{тоже сходится и при том сходится равномерно на $I=[a,b]$.}\\
        \text{3) в точке $x=a$ ряд сходится, т.к. $[a,b]\subset(x_0-R,x_0+R)$. Тогда по следствию теоремы}\\
        \text{о дифференцировании функциональных рядов из п.5 $\S$4 $\displaystyle\sum_{n=1}^{\infty}n\cdot a_n(x-x_0)^{n-1}=$}\\
        \text{$\displaystyle=\bigg(\sum_{n=0}^{\infty}a_n(x-x_0)^n\bigg)'$ $\forall x\in[a,b]$.}\\
        \text{Радиус сходимости ряда остаётся равным R: при дифференцировании радиус}\\
        \text{сходимости не уменьшается, но при обратном действии (интегрировании) радиус}\\
        \text{сходимости снова не уменьшается, но при этом вновь равен R.}}
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.7cm}
        \setcounter{equation}{0}
        \par{
        \textbf{Замечание: }\text{при дифференцировании сходимость на концах может исчезнуть.}\\
        \textbf{Пример: }\text{ряд $\displaystyle\sum_{n=1}^{\infty}(-1)^n\frac{x^n}{n}=-\ln{(1+x)}$ сходится при $x=1$, а производная от него}\\
        \text{$\displaystyle\sum_{n=1}^{\infty}(-1)^nx^{n-1}$ расходится при $x=1$. Радиус сходимости при этом не поменялся.}
        }

        \subsection*{5. Ряд Тейлора}
        \addcontentsline{toc}{subsection}{5. Ряд Тейлора}
        \par{\textbf{Пример для $e^x$:}\\
        \begin{equation*}
            f(x)=\sum_{n=0}^{\infty}\frac{x^n}{n!}, \ \ \ R=\infty
        \end{equation*}
        \begin{equation*}
            f'(x)=\sum_{n=1}^{\infty}\frac{x^{n-1}}{(n-1)!}=f(x)\Rightarrow\frac{f'(x)}{f(x)}=1\Rightarrow\frac{d[f(x)]}{f(x)dx}=1\Rightarrow\frac{d[f(x)]}{f(x)}=dx\Rightarrow
        \end{equation*}
        \begin{equation*}
            \Rightarrow\int\frac{d[f(x)]}{f(x)}=\int dx\Rightarrow \ln{f(x)}=x+c\Rightarrow f(x)=e^ce^x
        \end{equation*}
        \text{Пусть x = 0, тогда}
        \begin{equation*}
            \sum_{n=0}^{\infty}\frac{x^n}{n!}=f(x)=e^{x+c}\Rightarrow e^c=1\Rightarrow c=0\Rightarrow f(x)=e^x
        \end{equation*}
        }
        \newpage
        \par{\textbf{Ряды: }\text{Рассмотрим $\displaystyle\sum_{n=0}^{\infty}a_n(x-x_0)^n=f(x)$ и $R\ne0$, а ещё $\exists f^{(n)}$ $\forall n\in\mathbb{N}$.}\\
        \begin{equation*}
            f^{(k)}(x)=\sum_{n=k}^{\infty}a_nn(n-1)\ldots(n-k+1)(x-x_0)^{n-k}\Rightarrow f^{(k)}(x_0)=a_kk!\Rightarrow a_k=\frac{f^{(k)}(x_0)}{k!}
        \end{equation*}
        \text{Тогда $\displaystyle f(x)=\sum_{n=0}^{\infty}\frac{f^{(n)}(x_0)}{n!}(x-x_0)^n$. Рассмотрим функцию $f(x)\in C^{\infty}(\mathbb{R)}$, т.е. функцию,}\\
        \text{у которой производная любого порядка определена и непрерывна на всём $\mathbb{R}.$}\\
        \text{Куда сходится её ряд Тейлора?}\\
        \\
        \textbf{Пример:}
        \begin{equation*}
            f_0(x)=\begin{cases}
                {\large e^{-\frac{1}{(x-x_0)^2}}},  \ x\ne x_0\\
                0, \ \ \ \ \ \ \ \ \ \ \ x=x_0
            \end{cases}
        \end{equation*}
        \begin{equation*}
            \sum_{n=0}^{\infty}\frac{f_0^{(n)}(x_0)}{n!}(x-x_0)^n=0
        \end{equation*}
        \text{Т.о. ряд сходится при $\forall x\in\mathbb{R}\Rightarrow f+c_0f_0$ имеют один и тот же ряд Тейлора.}
        }
        \vspace{0.6cm}
        \par{
        \textbf{Теорема. } \text{Пусть $f$ имеет все производные в интервале $(x_0-R,x_0+R)$ и}\\
        \text{$|f^{(n)}(x)|\leq M$ $\forall n\in\mathbb{N}$ $\forall x\in(x_0-R,x_0+R)$. Тогда ряд Тейлора сходится к $f(x)$}\\
        \text{$\displaystyle \sum_{n=0}^{\infty}\frac{f^{(n)}(x_0)}{n!}(x-x_0)^n=f(x)$ $\forall x\in(x_0-R,x_0+R)$.}\\
        \textbf{Доказательство: }\text{по формуле Тейллора с остатком в интегральной форме}
        \begin{equation*}
            f(x)=\sum_{n=0}^{N}\frac{f^{(n)}(x_0)}{n!}(x-x_0)^n+{1\over{N!}}\int_{x_0}^{x} f^{(N+1)}(t)\cdot(x-t)^Ndt \text{ при }|x-x_0|<R
        \end{equation*}
        \text{По условию $|f^{(n)}(x)|\leq M$ $\forall n\in\mathbb{N}$ $\forall x\in(x_0-R,x_0+R)$.}
        \begin{itemize}
            \item \text{пусть $x>x_0$}\\
        \end{itemize}
        \begin{equation*}
            \bigg|f(x)-\sum_{n=0}^{N}\frac{f^{(n)}(x_0)}{n!}(x-x_0)^n\bigg|=\frac{1}{N!}\int_{x_0}^{x}\bigg|f^{(N+1)}(t)\bigg|\bigg|(x-t)^N\bigg|dt\leq
        \end{equation*}
        \begin{equation*}
            \leq\frac{1}{N!}\int_{x_0}^{x}\bigg|f^{(N+1)}(t)\bigg|\bigg|(\underbrace{x_0-t}_{\leq0})^N\bigg|dt=\frac{1}{N!}\int_{x_0}^{x}\bigg|f^{(N+1)}(t)\bigg|(t-x_0)^Ndt\leq\frac{M}{N!}\int_{x_0}^{x}(t-x_0)^Ndt=
        \end{equation*}
        \begin{equation*}
            \frac{M}{(N+1)!}(t-x_0)^{N+1}\Bigg|_{x_0}^{x}=\frac{M}{(N+1)!}(x-x_0)^{N+1}<\frac{M}{(N+1)!}R^{N+1}\xrightarrow{N\to\infty}0
        \end{equation*}
        \begin{itemize}
            \item \text{пусть $x<x_0$}\\
        \end{itemize}
        \begin{equation*}
            \bigg|f(x)-\sum_{n=0}^{N}\frac{f^{(n)}(x_0)}{n!}(x-x_0)^n\bigg|\leq\frac{1}{N!}\int_{x_0}^{x}\bigg|f^{(N+1)}(t)\bigg|(x_0-t)^Ndt\leq\frac{M}{N!}\int_{x_0}^{x}(x_0-t)^Ndt=
        \end{equation*}
        \begin{equation*}
            \frac{M}{(N+1)!}(x_0-t)^{N+1}\Bigg|_{x_0}^{x}=\frac{M}{(N+1)!}(x_0-x)^{N+1}<\frac{M}{(N+1)!}R^{N+1}\xrightarrow{N\to\infty}0
        \end{equation*}
        }

        \subsection*{1. Открытые и замкнутые множества в $\mathbb{R}^n$}
        \addcontentsline{toc}{subsection}{1. Открытые и замкнутые множества в $\mathbb{R}^n$}
        \par{\text{$\displaystyle<x,y>=\sum_{j=1}^{n}x_jy_j$, $\displaystyle<x,x>=\sum_{j=1}^{n}x_j^2\geq0$, $||x||=\sqrt{<x,x>}$ - длина (норма) $x$.}}
        \begin{enumerate}
            \item \par{\text{$<x,y>\leq||x||\cdot||y||$}}
            \item \par{\text{$||x||\geq0$, $||x||=0\Leftrightarrow x=0$}}
            \item \par{\text{$||\alpha x||=|\alpha|\cdot||x||$ \ \ $\forall x\in\mathbb{R}^n$, $\alpha\in\mathbb{R}$}}
            \item \par{\text{$||x+y||\leq||x||+||y||$, \ \ \ $\forall x,y\in\mathbb{R}^n$}}
        \end{enumerate}
            \par{\textbf{Def 1: }\text{Пусть $\delta>0$. Множество $B_{\delta}(x)=\{y\in\mathbb{R}^n:||x-y||<\delta\}$ называется}\\
            \text{открытым шаром радиуса $\delta$ с центром в точке $x$.}
        }
        \par{\textbf{Def 2: }\text{Множество $U\subset\mathbb{R}^n$ называется открытым, если $\forall x\in U$ $\exists\delta>0$ такое,}\\\text{что $B_{\delta}(x)\subset U$.}}
        \vspace{0.6cm}
        \par{\textbf{Примеры:}}
        \begin{wrapfigure}{r}{0.5\textwidth} % r - справа, l - слева; ширина
        \centering
        \includegraphics[width=0.25\textwidth]{rispar1gl2.png}
        \end{wrapfigure}
        \begin{enumerate}
            \item \par{$\mathbb{R}^n$ открыто}
            \item \par{$\emptyset$ тоже открыто}
            \item \par{$B_a(x)$ является открытым множеством}
            \item \par{Множество $\{y\in\mathbb{R}^n:||x-y||>a\}$ является открытым множеством}
            \item \par{\text{$\overline{B_a(x)}=\{y\in\mathbb{R}^n:||x-y||\leq a\}$ не является открытым, т.к. за точкой на сфере}\\\text{нет точек $\Rightarrow$ не найдётся такого шара из определения.}}
        \end{enumerate}
        \par{\textbf{Доказательство примера 3:} \text{$y\in B_a(x)$. Возьмём $\delta=a-||x-y||$ и пусть}\\
        \text{$z\in B_{\delta}(y)$. Проверим, что $B_{\delta}(y)\subset B_a(x):$ $||z-x||\leq||(z-y)+(y-x)||\leq$}\\
        \text{$\leq||z-y||+||y-x||<\delta+||y-x||=\delta+|-1|\cdot\underbrace{||x-y||}_{=a-\delta}=a$ $\Rightarrow z\in B_a(x)\Rightarrow$}\\
        \text{$\Rightarrow B_{\delta}(y)\subset B_a(x)$ $\forall y\in B_a(x)$. Тогда по определнию верно.}
        }
        \vspace{0.6cm}
        \par{\textbf{Def $2'$: } \text{$U\subset\mathbb{R}^n$ открыто $\Leftrightarrow \forall x\in U$ \ \ \ $\exists\delta>0, \ y\in U: \ \ x\in B_{\delta}(y)\subset U$.}
        }
        \par{\textbf{Утверждение: }\text{определения $2$ и $2'$ эквивалентны.}\\
        \textit{(доказательство следует из решения пр. 3 и из того, что можем взять весь шар)}
        }
        \vspace{0.6cm}
        \par{\textbf{Def 3: }\text{множество $F\subset\mathbb{R}^n$ называется замкнутым, если $\mathbb{R^n}\backslash F$ открыто.}
        }
        \vspace{0.4cm}
        \par{\textbf{Примеры:}}
        \begin{enumerate}
            \item \par{$\mathbb{R}^n\backslash\mathbb{R}^n=\emptyset$ (открыто) $\Rightarrow\mathbb{R}^n$ замкнуто.}
            \item \par{$\emptyset$ замкнуто, т.к. $\mathbb{R}^n\backslash\emptyset=\mathbb{R}^n$ открыто.}
            \item \par{\text{$\overline{B_a(x)}=\{y\in\mathbb{R}^n:||x-y||\leq a\}$ замкнуто.}}
            \item \par{\text{Сфера $S_a(x)=\{y\in\mathbb{R}^n:||x-y||= a\}$ замкнута.}}
        \end{enumerate}
        \par{\textbf{Def 4: }\text{проколотой окрестностью точки $x\in\mathbb{R}^n$ называется множество}\\
        \text{$\mathring{U}_{\delta}(x)=\mathring{B}_{\delta}(x)=\{y\in\mathbb{R}^n:0<||y-x||<\delta\}$.}}
        \vspace{0.5cm}
        \par{\textbf{Def 5: }\text{точка $x\in\mathbb{R}^n$ называется предельной точкой множества $M\subset\mathbb{R}^n$, если}\\
        \text{$\forall\delta>0$ $\mathring{U}_{\delta}(x)\cap M\neq\emptyset$.}
        }
        \vspace{0.4cm}
        \par{\textbf{Примеры:}}
        \begin{enumerate}
            \item \par{\text{$M=(0,1)\subset\mathbb{R}$. $[0,1]\subset\mathbb{R}$ - множество предельных точек.}}
            \item \par{\text{$M=(0,1)\cup\{2\}\subset\mathbb{R}$. $[0,1]\subset\mathbb{R}$ - множество предельных точек.}}
        \end{enumerate}
        \par{\text{Из примеров можно сделать выводы, что предельная точка может не лежать в M,}\\
        \text{а ещё, что не все точки из M являются предельными.}}
        \vspace{0.6cm}
        \par{\textbf{Теорема (критерий замкнутости)}\\
        \text{множество $M\subset\mathbb{R}^n$ замкнуто $\Leftrightarrow$ оно содержит все свои предельные точки.}\\
        \textbf{Доказательство: $[\Rightarrow]$ }\text{$M$ замкнуто:}\\
        \text{$M$ замкнуто и точка $x\not\in M$ $\Rightarrow x\in\mathbb{R}^n\backslash M$ открыто $\Rightarrow\exists\delta>0:$ $B_{\delta}(x)\subset\mathbb{R}^n\backslash M\Rightarrow$}\\
        \text{$\Rightarrow\mathring{U}_{\delta}(x)\cap M=\emptyset$ $\Rightarrow x$ не является предельной точкой.}\\
        \text{$[\Leftarrow]$ $M$ содержит все свои предельные точки:}\\
        \text{рассмотрим $x\in\mathbb{R}^n\backslash M\Rightarrow$ $x$ не является предельной точкой для $M$, т.е. $\exists\delta>0:$}\\
        \text{$\mathring{U}_{\delta}(x)\cap M=\emptyset$ $\Rightarrow$ $B_{\delta}(x)\cap M=\emptyset\Rightarrow B_{\delta}(x)\subset\mathbb{R}^n\backslash M$ открыто, тогда $М$ замкнуто.}
        }

        \subsection*{2. Сходимость в $\mathbb{R}^n$}

        \par{\ }
        \par{\textbf{Def 6: }\text{пусть $\{x_k\}$ - последовательность в $\mathbb{R}^n$} и $a\in\mathbb{R}^n$ - фиксированная точка.}\\
        \text{Предел последовательности $\{x_k\}$ равен $a$, если $\displaystyle\lim_{n\to\infty}||x-a||=0$.}
        \vspace{0.4cm}
        \par{\textbf{Def 7: }\text{последовательность $\{x_k\}\subset\mathbb{R}^n$ называется фундаментальной, если}\\
        \text{$\forall\epsilon>0$ $\exists N\in\mathbb{N}: \forall n,m>N$ $||x_n-x_m||<\epsilon$.}
        }
        \newpage
        \par{\textbf{Теорема (критерий Коши)}\\
        \text{Последовательность имеет предел $\Leftrightarrow$ она фундаментальна.}\\
        \textbf{Доказательство: }\text{пусть $x_{k_i}$ - i-ая координата элемента $x_k$}\\
        \textbf{[$\Rightarrow$] } \textit{последовательность имеет предел}\\
        \text{Распишем определение предела: $\forall\epsilon>0$ $\exists N\in\mathbb{N}:$ $\forall k>N$ $||x_k-a||<\epsilon$, подставим:}\\
        \text{$||x_k-x_m||\leq||x_k-a+a-x_m||\leq||x_k-a||+||x_m-a||<2\epsilon$}\\
        \textbf{[$\Leftarrow$] }\textit{последовательность фундаментальна}\\
        \text{$||x_{k_i}-x_{m_i}||\leq||x_k-x_m||\to0$, т.к. последовательность фундаментальна, тогда}\\
        \text{$||x_{k_i}-x_{m_i}||\to0\Rightarrow$ последовательность $\{x_{k_i}\}$ тоже фундаментальна $\displaystyle\Rightarrow\exists\lim_{k\to\infty}x_{k_i}=x_i$.}\\
        \text{Рассмотрим $x=(x_1,\ldots,x_n)$: фикс. $\epsilon>0$ и $\forall i=1,\ldots,n$ выберем $N_i(\epsilon)\in\mathbb{N}$ такое, что}\\
        \text{$||x_{k_i}-x_{m_i}||<\frac{\epsilon}{\sqrt{n}}$ $\forall k>N$. Знаем, что $|x_i|\leq||x||$, тогда выберем $N=\max\{N_1,\ldots,N_n\}$,}\\
        \text{тогда при $k>N$ верно:}
        \begin{equation*}
            ||x_k-x||=\bigg(\sum_{i=1}^{n}(x_{k_i}-x_i)^2\bigg)^{1\backslash2}\leq\bigg(\sum_{i=1}^{n}(\frac{\epsilon}{\sqrt{n}})^2\bigg)^{1\backslash2}=\epsilon
        \end{equation*}
        }

        \subsection*{3. Предел функции нескольких переменных}
        \addcontentsline{toc}{subsection}{3. Предел функции нескольких переменных}
        \par{\ }
        \par{\textbf{Def 8: }Пусть $f$ задана на $V\subset\mathbb{R}^n$ и принимает значения из $\mathbb{R}$, $a$ - предельная точка \\ множества $V\subset\mathbb{R}^n$, $l\in\mathbb{R}$. Говорят, что \underline{$f$ имеет предел в точке $a$ равный $l$}, и пишут \\ $\displaystyle\lim_{x\to a}f(x)=l$, если $\forall\epsilon>0$ $\exists\delta>0: x\in V\cap\mathring{U}_{\delta}(a) \ \ |f(x)-l|<\epsilon$.}
        \vspace{0.3cm}
        \par{\textbf{Def $8'$: }Пусть $f$ задана на $V\subset\mathbb{R}^n$ и принимает значения из $\mathbb{R}$, $a$ - предельная точка \\ множества $V\subset\mathbb{R}^n$, $l\in\mathbb{R}$ (те же, что и в опр. 8). Предел функции $f$ в точке $a$ существует\\ и равен $l$ $\Leftrightarrow\forall\{x_k\}\subset V, x_k\ne a, x_k\to a$ $\displaystyle\exists\lim_{k\to\infty}f(x_k)=l$.}
        \vspace{0.3cm}
        \par{\textbf{Утверждение: }определения 8 и $8'$ эквивалентны.\\
        \textbf{Доказательство: } \\
        \textbf{[$\Rightarrow$]} $\displaystyle\exists\lim_{x\to a}f(x)=l$ и $x_k\to a$\\
        Фикс. $\epsilon>0$ и выберем $\delta>0:\forall x\in V\cap\mathring{U}_{\delta}(a)$ $|f(x)-l|<\epsilon$. Т.к. $x_k\to a$, то $\exists N\in\mathbb{N}:\forall k>N$ $x_k\subset V\cap\mathring{B}_{\delta}(a)$, тогда $|f(x_k)-l|<\epsilon$.\\
        \textbf{[$\Leftarrow$]} (от противного)\\
        Пусть $\forall\{x_k\}\subset V\backslash\{a\}$, $x_k\to a$ верно, что $\displaystyle\lim_{k\to\infty}f(x_k)=l$, но $\displaystyle\lim_{x\to a}f(x)\ne l$. Тогда $\displaystyle\exists\epsilon_0>0:\forall\delta>0 \ \ \exists x\in V\cap\mathring{B}_{\delta}(a)$, в которой $|f(x)-l|\geq\epsilon_0$. Рассмотрим $\delta={1\over k}$ и $x_k$ - соотв. точка из $V\cap\mathring{B}_{1\over k}(a)$, тогда $\displaystyle||x_k-a||<{1\over k}\xrightarrow[k\to\infty]{}0$, т.е. $x_k\to a$, но $|f(x_k)-l|\geq\epsilon_0$, т.е. $\displaystyle\lim_{k\to\infty}f(x_k)\ne l$. Тогда приходим к $\lightning$ с предположением.
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.6cm}
        \par{\textbf{Пусть $n=2$} и пусть существуют пределы $\displaystyle\lim_{(x,y)\to(x_0,y_0)}f(x,y)$, $\displaystyle\lim_{y\to y_0}\lim_{x\to x_0}f(x,y)$ и \\$\displaystyle\lim_{x\to x_0}\lim_{y\to y_0}f(x,y)$, тогда они равны. Но если существуют только два из них, то про третий ничего сказать нельзя.}
        \newpage
        \par{\textbf{Примеры:}}
        \begin{enumerate}
            \item \par{$f(x,y)=\begin{cases}
                x\sin{1\over y}, \ y\ne0 \\
                0 \ \ \ \ \ \ \ , \ y=0
            \end{cases}$
            $\displaystyle\lim_{(x,y)\to(0,0)}f(x,y)=0$, т.к. $x\to0$, а $|\sin{1\over y}|\leq1$, ещё \\
            $\displaystyle\exists\lim_{y\to0}\lim_{x\to0}f(x,y)=0$, но $\displaystyle\nexists\lim_{x\to0}\lim_{y\to0}f(x,y)$, т.к. $\displaystyle\nexists\lim_{y\to0}\sin{1\over y}$.}
            \item \par{$f(x,y)=\begin{cases}
                \displaystyle\frac{xy}{x^2+y^2}, \ x^2+y^2\ne0 \\
                \ \ \ \ \ 0 \ \ \ \ , \ x=y=0
            \end{cases}$ У такой функции существуют оба повторных предела, но не существует двойного предела, т.к. определение $8'$ не выполняется ($f(0,{1\over k})\to0 \ne {1\over2}\xleftarrow{}f({1\over k}, {1\over k})$).}
        \end{enumerate}
        \vspace{0.6cm}
        \par{\textbf{Следствие: }Пусть сущетсвуют пределы $\displaystyle\lim_{x\to a}f(x)=l$ и $\displaystyle\lim_{x\to a}g(x)=m$, тогда существуют и пределы функций $f\pm g$, $f\cdot g$, $\displaystyle\frac{f}{g}$ при $m\ne0$, равные соотв. $l\pm m$, $l\cdot m$, $\displaystyle\frac{l}{m}$. Доказательство следует из определения $8'$.}
        \vspace{0.7cm}
        \par{\textbf{Def 9: }Множество $V\subset\mathbb{R}^n$ называется ограниченым, если $\exists c\geq0:||x||\leq c$ $\forall x\in V$. Последовательность $\{x_k\}\subset\mathbb{R}^n$ называется ограниченной, если множество $\{x_k, \ k\in\mathbb{N}\}$ ограничено, т.е. $\exists c\geq0: ||x_k||\leq c \ \ \ \forall k\in\mathbb{N}$.}
        \vspace{0.4cm}
        \par{\textbf{Теорема }Из любой ограниченной последовательности можно выделить сходящуюся подпоследовательность. \\
        \textbf{Доказательство: }докажем по индукции. База $n=1$ (теорема Больцано-Вейерштрасса). \\
        Пусть в $\mathbb{R}^n$ можно выделить сходящуюся подпоследовательность. Рассмотрим $\{x_k\}\subset\mathbb{R}^{n+1}$. $x_k=(x_k',x_{k,n+1})$, т.е. добавили координату $n+1$. Т.к. $|x_{k,n+1}|\leq||x_k||\leq c$ и\\ $||x_k'||\leq||x_k||\leq c$, то обе последовательности $\{x_{k,n+1}\}$ и $\{x_k'\}$ ограничены. По индукционному предположению из $\{x_k'\}$ можно выделить сходящуюся подпоследовательность\\ $\{x_{k_m}'\}\xrightarrow{m\to\infty}x'$. Воспользовавшись базой индукции, выделим из последовательности $\{x_{k_m,n+1}\}$ (она ограничена) сходящуюся подпоследовательность $\{x_{k_{m_l} \ ,n+1}\}\to x_{n+1}$. Теперь рассмотрим последовательность $\{x_{k_{m_l}}', \ x_{k_{m_l} , \ n+1}\}\to\{x',x_{n+1}\}$. 
        }

        \subsection*{4. Непрерывность}
        \addcontentsline{toc}{subsection}{4. Непрерывность}
        \par{\ }
        \par{\textbf{Def 10: }Пусть $f$ задана на $V\subset\mathbb{R}^n$ и принимает вещественные значения, $a\in V$ является предельной точкой $V$. Функция $f$ непрерывна в точке $a$, если $\displaystyle\lim_{x\to a}f(x)=f(a)$. \\
        Функция $f$ непрерывна на множестве $A\subset V$, если $f$ непрерывна в любой точке из этого множества.
        }
        \vspace{0.4cm}
        \par{\textbf{Примеры:}}
        \begin{enumerate}
            \item \par{$f(x)=||x||$ непрерывна. $\bigg|||x||-||a||\bigg|\leq||x-a||\xrightarrow{x\to a}0$.}
            \item \par{$f(x,y)=\begin{cases}
                \displaystyle\frac{xy}{x^2+y^2}, \ x^2+y^2\ne0 \\
                \ \ \ \ \ 0 \ \ \ \ , \ x=y=0
            \end{cases}$ разрывна, т.к. в точке (0,0) не существует двойного предела.}
        \end{enumerate}
        \newpage
        \par{\textbf{Def 11: }Пусть функция $f$ задана на множестве $V\subset\mathbb{R}^n$. $f$ равномерно непрерывна на $[a,b]$, если $\forall\epsilon>0 \ \exists\delta>0:\forall x,x'\in V \ \ ||x-x'||<\delta \Rightarrow |f(x)-f(x')|<\epsilon$.}
        \vspace{0.3cm}
        \par{\textbf{Теорема (Кантор)}\\
        Пусть $V\subset\mathbb{R}^n$ - замкнутое и ограниченное множество, а $f$ непрерывна на $V$, тогда $f$ равномерно непрерывна на $V$.\\
        \textbf{Доказательство: }Пусть $V$ замкнуто и ограничено, а $f$ непрерывна на нём, но не равномерно. Т.е. $\exists\epsilon_0>0:\forall\delta>0 \ \exists x,x'\in V, ||x-x'||<\delta$, но $|f(x)-f(x')|\geq\epsilon_0$. Рассмотрим $\delta_k={1\over k}$, $k\in\mathbb{N}$: для них $\exists x_k,x_k'$ такие, что $||x_k-x_k'||<\delta={1\over k}$, но $|f(x_k)-f(x_k')|>\epsilon$. $\{x_k\}\subset V$ ограничена, т.к. $V$ органичено, тогда существует сходящаяся подпоследовательность $x_{k_m}\to x$. Т.к. $V$ замкнуто, то $x\in V$. Рассмотрим $x_{k_m}'$:
        \begin{equation*}
            ||x_{k_m}'-x||=||x_{k_m}'-x_{k_m}+x_{k_m}-x||\leq||x_{k_m}'-x_{k_m}||+||x_{k_m}-x||\to0
        \end{equation*}
        Тогда $x_{k_m}'\to x$. Т.к. $f$ непрерывна на $V$, то она непрерывна в точке $x$, тогда
        \begin{equation*}
        \displaystyle\lim_{m\to\infty}f(x_{k_m}')=\lim_{m\to\infty}f(x_{k_m})=f(x)
        \end{equation*}
        Но вернёмся к равномерной сходимости: $|f(x_{k_m})-f(x_{k_m}')|\geq\epsilon_0$. Мы выяснили, что левая часть стремится к нулю. Тогда $0\geq\epsilon_0>0$, что приводит нас к $\lightning$.
        }
        \par{{\raggedleft ч.т.д.\par}}
        \vspace{0.6cm}
        \par{\textbf{Теорема (Вейерштрасс)}\\
        Пусть $V\subset\mathbb{R}^n$ замкнутое и ограниченное множество, а $f$ непрерывна на $V$, тогда $f$ ограничена на $V$ и достигает своего наибольшего и наименьшего значения. Т.е. $\exists a_1,a_2\in V: f(a_1)\leq f(x)\leq f(a_2)$.\\
        \textbf{Доказательство: }пусть $f$ неограничена сверху, т.е. $\forall k\in\mathbb{N}$ $\exists x_k\in V:f(x_k)\geq k$. $\{x_k\}\subset V$ ограничена, т.к. множество ограничено, тогда $\exists x_{k_m}$, которая сходится. Но т.к. $f(x_{k_m})\geq k_m$, то $\displaystyle\lim_{m\to\infty}f(x_{k_m})=+\infty$, что приводит нас к $\lightning$, т.к. $\displaystyle\lim_{m\to\infty}f(x_{k_m})=f(x)$ в силу непрерывности. Ну тогда функции придётся быть ограниченной сверху на $V$. Рассмотрим функцию $-f$: она ограничена сверху (доказали), тогда $f$ ограничена снизу.\\
        Т.к. $f$ ограничена сверху, то $A=\sup f(x)$ $x\in V$ конечен. По определению sup $\forall k\in\mathbb{N}$ $\exists x_k\in V:$ $f(x_k)>A-{1\over k}$. $\{x_k\}\subset V$ ограничена, т.к. множество ограничено, тогда $\exists x_{k_m}\to x\in V$ и $\displaystyle\lim_{m\to\infty}f(x_{k_m})=f(x)$ в силу непрерывности. Тогда выпишем неравенство:
        \begin{equation*}
            A-{1\over k}<f(x_{k_m})\leq A
        \end{equation*}
        А в пределе получим
        \begin{equation*}
            A\leq f(x)\leq A
        \end{equation*}
        Тогда $f(x)=A$, т.е. sup является значением.
        }

        \section*{$\S$2. Дифференциальное исчисление функций нескольких переменных}

        \subsection*{1. Дифференцируемость}
        \addcontentsline{toc}{subsection}{1. Дифференцируемость}
        \par{\ }
        \par{\textbf{Def 1: }Пусть $f$ задана на $V\subset\mathbb{R}^n$ и $x\in V$ лежит в $V$ вместе с некоторым шаром $B_{\delta}(x)$. Функция $f:V\to\mathbb{R}$ называется дифференцируемой в точке $x$, если приращение $\Delta f(x,\Delta x)=f(x+\Delta x)-f(x)$ допускает представление $\Delta f(x,\Delta x)=A_1\Delta x_1+\ldots+A_n\Delta x_n+\bar{\bar{o}}(\Delta x)$ при $\Delta x\to0$.}
        \vspace{0.4cm}
        \par{\textbf{Замечание: }если $f$ дифференцируема в точке $x$, то она непрерывна в точке $x$.\\\textbf{Обратное неверно! Пример: } $f(x)=||x||$ в точке $x=0$ данная функция непрерывна, но не дифференцируема.\\ \textbf{Пусть $x\neq0:$}
        \begin{equation}
            ||x+\Delta x||-||x||=\frac{||x+\Delta x||^2-||x||^2}{||x+\Delta x||+||x||}
        \end{equation}
        Рассмотрим $||x+\Delta x||^2-||x||^2$:
        \begin{equation*}
            ||x+\Delta x||^2-||x||^2=\sum_{i=1}^{n}(x_i+\Delta x_i)^2-\sum_{i=1}^{n}x_i^2=2\sum_{i=1}^nx_i\Delta x_i+\sum_{i=1}^n\Delta x_i^2=
        \end{equation*}
        \begin{equation}
            =2\sum_{i=1}^nx_i\Delta x_i+||\Delta x||^2=2\sum_{i=1}^nx_i\Delta x_i+\bar{\bar{o}}(\Delta x)
        \end{equation}
        Рассмотрим $\frac{1}{||x+\Delta x||+||x||}$ \textit{(честно-честно это так)}:
        \begin{equation}
            \frac{1}{||x+\Delta x||+||x||}=\frac{1}{2||x||}+\bar{\bar{o}}(\Delta x)
        \end{equation}
        Подставим сейчас (2) и (3) в (1):
        \begin{equation}
            (1)=\sum_{i=1}^n\frac{x_i}{||x||}\Delta x_i + \bar{\bar{o}}(x)
        \end{equation}
        Тогда из (4) следует, что функция дифференцируема в точках $x\neq0$.\\
        \textbf{Пусть $x=0:$ }, тогда $||x+\Delta x||-||x||=||\Delta x||=A_1\Delta x_1+\ldots+A_n\Delta x_n+\bar{\bar{o}}(\Delta x)$. Рассмотрим случаи, когда есть приращение только по координате с номером $k$, т.е. $\Delta x_i=0$ при $i\ne k$:
        \begin{equation}
            ||\Delta x||=||\Delta x_k||=A_k\Delta x_k+\bar{\bar{o}}(\Delta x_k)
        \end{equation}
        Тогда из (5) получаем, что
        \begin{equation}
            1=A_k\frac{\Delta x_k}{||\Delta x_k||}+\bar{\bar{o}}(1)
        \end{equation}
        Тогда предела правой части нет, ведь тогда пределы справа и слева должны совпадать, но они имеют разные знаки. Это значит, что $f(x)=||x||$ не дифференцируема в нуле.
        }
        \newpage
        \subsection*{2. Частные производные}
        \addcontentsline{toc}{subsection}{2. Частные производные}
        \par{Пусть $k$ фикс. $(k=1,\ldots,n)$ и пусть $\Delta x_i=0$ $\forall i\ne k$. \\
        Обозначим за $\Delta f(x, \Delta x_k)=f(x_1, \ldots,x_k+\Delta x_k,\ldots,x_n)-f(x_1,\ldots,x_n)$.\\ \\
        Частной производной функции $f$ по переменной $x_k$ называется предел (если он существует) $\displaystyle\frac{\partial f(x)}{\partial x_k}=\lim_{\Delta x_k\to0}\frac{\Delta f(x,\Delta x_k)}{\Delta x_k}$ (все обозначения введены выше).
        }
        \vspace{0.6cm}
        \par{\textbf{Замечание:}\\
        Пусть $f$ дифференцирована в точке $x$, тогда $\Delta f(x,\Delta x_k)=A_k\Delta x_k+\bar{\bar{o}}(\Delta x_k)$, тогда $\frac{\Delta f(x)}{\Delta x_k}=A_k+\bar{\bar{o}}(1)\Rightarrow\frac{\partial f(x)}{\partial x_k}=A_k$.
        }
        \setcounter{equation}{0}
        \vspace{0.2cm}
        \par{\textbf{Обратное неверно! Пример:}\\
        $f(x,y)=\begin{cases}
                \displaystyle\frac{xy}{x^2+y^2}, \ x^2+y^2\ne0 \\
                \ \ \ \ \ 0 \ \ \ \ , \ x=y=0
            \end{cases}$ разрывна в точке $(0,0)\Rightarrow$ не дифференцируема, но при этом обе её частные производные равны 0.}
        \vspace{0.8cm}
        \par{\textbf{Теорема. }Пусть $f$ имеет частные производные $\displaystyle\frac{\partial f}{\partial x_i}$ $\forall i=1,\ldots,n$ в некотором шаре $B_{\delta}(x)$, причём $\displaystyle\frac{\partial f}{\partial x_i}$ непрерывна в точке $x$. Тогда $f$ дифференцируема в точке $x$.\\
        \textbf{Доказательство: }$\Delta f(x,\Delta x)=f(x+\Delta x)-f(x)=\bigg(f(x_1+\Delta x_1,\ldots,x_n+\Delta x_n)-$\\$-f(x_1,x_2+\Delta x_2,\ldots,x_n+\Delta x_n)\bigg)+\ldots+\bigg(f(x_1,\ldots,x_{n-1},x_n+\Delta x_n)-f(x_1,\ldots,x_n)\bigg).$\\
        Обозначим
        \begin{equation}
            \frac{\partial f}{\partial x_i}(\theta_i)=\frac{\partial f}{\partial x_i}(x_1,\ldots,x_{i-1},x_i+\theta_i x_i,x_{i+1}+\Delta x_{i+1},\ldots,x_n+\Delta x_n)
        \end{equation}
        Тогда по теореме Лагранжа для функции одной переменной\\
        \textit{Напоминание: теорема Лагранжа для функции одной переменной гласит, что если эта ф-ция замкнута на $[a,b]$ и имеет конечную $f'(x)$ $\forall x\in(a,b)$, то $\exists c\in(a,b): \frac{f(b)-f(a)}{b-a}$.}
        \begin{equation}
            \Delta f(x,\Delta x)=\frac{\partial f}{\partial x_1}(\theta_1)\Delta x_1+\ldots+\frac{\partial f}{\partial x_n}(\theta_n)\Delta x_n
        \end{equation}
        Добавим и вычтем $\displaystyle\frac{\partial f(x)}{\partial x_i}\Delta x_i$:
        \begin{equation}
            (2)=\frac{\partial f(x)}{\partial x_1}\Delta x_1+\bigg(\frac{\partial f}{\partial x_1}(\theta_1)-\frac{\partial f(x)}{\partial x_1}\bigg)\Delta x_1+\ldots+\frac{\partial f(x)}{\partial x_n}\Delta x_n+\bigg(\frac{\partial f}{\partial x_n}(\theta_n)-\frac{\partial f(x)}{\partial x_n}\bigg)\Delta x_n
        \end{equation}
        В силу непрерывности частных производных делаем вывод, что
        \begin{equation}
            \bigg(\frac{\partial f}{\partial x_1}(\theta_1)-\frac{\partial f(x)}{\partial x_1}\bigg)\Delta x_1+\ldots+ \bigg(\frac{\partial f}{\partial x_n}(\theta_n)-\frac{\partial f(x)}{\partial x_n}\bigg)\Delta x_n=\bar{\bar{o}}(\Delta x)
        \end{equation}
        Тогда, применив (4) к (3), получим $\displaystyle\Delta f(x,\Delta x)=\sum_{i=1}^n\frac{\partial f(x)}{\partial x_i}\Delta x_i+\bar{\bar{o}}(\Delta x)$. Таким образом функция действительно дифференцируема в точке $x$.
        }

        \subsection*{3. Производная сложной функции}
        \addcontentsline{toc}{subsection}{3. Производная сложной функции}
        \par{\ }
        \par{\textbf{Теорема. }Пусть функции $g_1,\ldots,g_m$ дифференцируемы в точке $x\in\mathbb{R}^n$, а функция $f$ дифференцируема в точке $y=\bigg(g_1(x),\ldots,g_m(x)\bigg)\in\mathbb{R}^m$. Тогда сложная функция \\ $x\mapsto f\bigg(g_1(x),\ldots,g_m(x)\bigg)$ дифференцируема в точке $x$, причём
        \begin{equation*}
            \frac{\partial}{\partial x_k}f(g_1(x),\ldots,g_m(x))=\sum_{i=1}^m\frac{\partial f(g_1(x),\ldots,g_m(x))}{\partial g_i}\frac{\partial g_i}{\partial x_k}
        \end{equation*}
        \textbf{Доказательство: }т.к. $f$ дифференцируема в точке $y$, то 
        \begin{equation}
            \Delta f(y,\Delta y)=\sum_{i=1}^m\frac{\partial f(y)}{\partial y_i}\Delta y_i+\bar{\bar{o}}(\Delta y)
        \end{equation}
        Рассмотрим $y_i=g_i(x)$ и $\Delta y_i$:
        \begin{equation}
            \Delta y_i=\Delta g_i(x,\Delta x)=\sum_{k=1}^n\frac{\partial g_i(x)}{\partial x_k}\Delta x_k+\bar{\bar{o}}(\Delta x)
        \end{equation}
        Докажем, что $\bar{\bar{o}}(\Delta y)=\bar{\bar{o}}(\Delta x)$:
        \begin{equation}
            \frac{\bar{\bar{o}}(\Delta y)}{||\Delta x||}=\underbrace{\frac{\bar{\bar{o}}(\Delta y)}{||\Delta y||}}_{\bar{\bar{o}}(1)}\cdot\frac{||\Delta y||}{||\Delta x||}
        \end{equation}
        Осталось рассмотреть частное норм $||\Delta y||$ и $||\Delta x||$:
        \begin{equation}
            \bigg|\frac{\Delta y_i}{||\Delta x||}\bigg|=\bigg|\sum_{k=1}^n\frac{\partial g_i(x)}{\partial x_k}\frac{\Delta x_k}{||\Delta x||} +\frac{\bar{\bar{o}}(\Delta x)}{||\Delta x||}\bigg|\leq\sum_{k=1}^n\underbrace{\bigg|\frac{\partial g_i(x)}{\partial x_k}\bigg|}_{const}\cdot\underbrace{\frac{|\Delta x_k|}{||\Delta x||}}_{\leq1}+\underbrace{|\bar{\bar{o}}(1)|}_{\textit{ограничена}}
        \end{equation}
        Следовательно вся конструкция (4) ограничена, тогда и частное норм $||\Delta y||$ и $||\Delta x||$, состоящее из конструкций (4), тоже ограничено. Тогда из (3) $\Rightarrow$ $\bar{\bar{o}}(\Delta y)=\bar{\bar{o}}(\Delta x)$.\\
        Теперь в (1) подставим (2) и $\bar{\bar{o}}(\Delta y)=\bar{\bar{o}}(\Delta x)$ и получим:
        \begin{equation*}
            \Delta f[g_1(x),\ldots,g_m(x), \Delta x]=f[g_1(x+\Delta x),\ldots,g_m(x+\Delta x)]-f[g_1(x),\ldots,g_m(x)]=
        \end{equation*}
        \begin{equation*}
            =\sum_{i=1}^m\frac{\partial f[g_1(x),\ldots,g_1(x)]}{\partial g_i}\cdot\sum_{k=1}^n\frac{\partial g_i(x)}{\partial x_k}\Delta x_k+\bar{\bar{o}}(\Delta x)=
        \end{equation*}
        \begin{equation*}
            =\sum_{i=1}^m\sum_{k=1}^n\frac{\partial f[g_1(x),\ldots,g_1(x)]}{\partial g_i}\cdot\frac{\partial g_i(x)}{\partial x_k}\Delta x_k+\bar{\bar{o}}(\Delta x)=
        \end{equation*}
        \begin{equation}
            =\sum_{k=1}^n\underbrace{\Bigg(\sum_{i=1}^m\frac{\partial f[g_1(x),\ldots,g_1(x)]}{\partial g_i}\cdot\frac{\partial g_i(x)}{\partial x_k}\Bigg)}_{\frac{\partial}{\partial x_k}f(g_1(x),\ldots,g_m(x))}\Delta x_k+\bar{\bar{o}}(\Delta x)
        \end{equation}
        В (5) получили разложение по $\Delta x_k$, где все значки имеют смысл (по условию), поэтому наша функция дифференцируема в точке $x$.
        }

        \subsection*{4. Производная по направлению. Градиент}
        \addcontentsline{toc}{subsection}{4. Производная по направлению. Градиент}
        \par{\ }
        \par{\textbf{Def 2: } пусть $f$ определена на $V\subset\mathbb{R}^n$, $x\in B_{\delta}(x)\subset V$, $l\in\mathbb{R}^n$, $||l||=1$, $t\in\mathbb{R}$. \\
        Производной по направлению $\frac{\partial f(x)}{\partial l}$ называется величина
        \begin{equation*}
            \frac{d}{dt}f(x+lt)\bigg|_{t=0}=\lim_{\Delta t\to0}\frac{f(x+l\Delta t)-f(x)}{\Delta t}
        \end{equation*}
        }
        \vspace{0.5cm}
        \par{\textbf{Теорема.} Пусть $f$ дифференцируема в точке $x$ и пусть $l\in\mathbb{R}^n$, $||l||=1$, тогда $\exists\frac{\partial f(x)}{\partial l}$ и справедлива формула
        \begin{equation*}
            \frac{\partial f(x)}{\partial l}=\sum_{i=1}^n\frac{\partial f(x)}{\partial x_i}l_i
        \end{equation*}
        \textbf{Доказательство: }$k=1$, $x_1=t$, $g_i(t)=x_i+t\cdot l_i$
        \begin{equation*}
            \frac{d}{dt}f(x+lt)=\sum_{i=1}^n\frac{\partial f(x+lt)}{\partial x_i}\cdot\frac{d}{dt}(x_i+l_it)=\sum_{i=1}^n\frac{\partial f(x+lt)}{\partial x_i}\cdot l_i
        \end{equation*}
        И при $t=0$ получим формулу.
        }
        \par{{\raggedleft ч.т.д.\par}}
        \noindent\hrulefill\noindent
        \par{\textbf{Пример:} \textit{(нельзя избавиться от дифференцируемости)} \\
        \begin{equation*}
            f(x,y)=\sqrt[3]{x^2y} \ \ \ \ \ l=(l_x,l_y), \ l_x^2+l_y^2=1
        \end{equation*}
        Рассмотрим в точке $(0,0)$:
        \begin{equation*}
            \lim_{\Delta t\to0}\frac{f(0+l_x\Delta t, \ 0+l_y\Delta t)-f(0,0)}{\Delta t}=\lim_{\Delta t\to0}\frac{\sqrt[3]{l_x^2l_y\cdot\Delta t^3}}{\Delta t}=\sqrt[3]{l_x^2l_y}\neq0
        \end{equation*}
        Рассмотрим частные производные:
        \begin{equation*}
            \frac{\partial f(0,0)}{\partial x}=\frac{\partial f(0,0)}{\partial y}=0
        \end{equation*}
        Тогда формула $\displaystyle\frac{\partial f(x)}{\partial l}=\sum_{i=1}^n\frac{\partial f(x)}{\partial x_i}l_i$ не работает, т.к. $0\neq\sqrt[3]{l_x^2l_y}=0\cdot l_x+0\cdot l_y=0$. Причина заключается в том, что функция не дифференцируема.}
        \par
        \noindent\hrulefill\noindent
        \par{\ }
        \par{\textbf{Градиент: } $\displaystyle\nabla f(x)=\bigg(\frac{\partial f(x)}{\partial x_1},\ldots,\frac{\partial f(x)}{\partial x_n}\bigg)$ и $\displaystyle\frac{\partial f(x)}{\partial l}=<\nabla f(x), \ l>=||\nabla f(x)||\cdot\cos{\phi}$.\\
        Градиент — это векторная величина, показывающая направление наискорейшего изменения (роста) какой-либо величины.}

        \subsection*{5. Формула Лагранжа}
        \addcontentsline{toc}{subsection}{5. Формула Лагранжа}
        \par{\ }
        \par{\textbf{Теорема (Лагранж)}\\
        Пусть $f$ задана и непрерывна на $V\subset\mathbb{R}^n$ и $x_1,x_0\in V$ вместе с $I=\{y\in\mathbb{R}^n:y=x_0+t(x-x_0), t\in[0,1]\}$. Пусть ещё $f$ дифференцируема во всех внутренних точках $I$. Тогда $\exists c\in I:f(x)-f(x_0)=<\nabla f(c),x-x_0>$.\\
        \textbf{Доказательство: }Рассмотрим $g(t)=f(x_0+t(x-x_0))$, она непрерывна на $[0,1]$ и дифференцируема $\forall t\in(0,1)$ в силу соответсвующих свойств функции $f$. По теореме Лагранжа для функции одной переменной (а наша $g$ является таковой) $\exists\theta\in(0,1):$
        \begin{equation}
            \underbrace{g(1)}_{=f(x)}-\underbrace{g(0)}_{=f(x_0)}=g'(\theta)=\sum_{i=1}^n\frac{\partial f(x_0+\theta(x-x_0))}{\partial x_i}\cdot(x_i-x_{i_0})
        \end{equation}
        Тогда из (1) получим, что $f(x)-f(x_0)=<\nabla f(c),x-x_0>$ и $c=x_0+\theta(x-x_0)$.
        }

        \subsection*{6. Производные высших порядков}
        \addcontentsline{toc}{subsection}{6. Производные высших порядков}
        \setcounter{equation}{0}
        \par{\ }
        \par{\textbf{Def 3: }$\displaystyle\frac{\partial^mf(x)}{\partial x_{i_1}\cdot\ldots\cdot\partial x_{i_m}}=\frac{\partial}{\partial x_1}\bigg(\frac{\partial^{m-1}f(x)}{\partial x_{i_2}\cdot\ldots\cdot\partial x_{i_m}}\bigg)$ - частная производная порядка $m$ (определение по индукции). В данном случае $x_{i_k}$ может быть любым.}
        \par
        \noindent\hrulefill\noindent
        \par{\ }
        \par{\textbf{Пример: }\textit{(существует проблема с порядком дифференцирования)}
        \begin{equation*}
            f(x,y)=\begin{cases}
                xy\frac{x^2-y^2}{x^2+y^2}, \ x^2+y^2\ne0 \\
                0 \ \ \ \ \ \ \ \ \ , \ x=y=0
            \end{cases}
        \end{equation*}
        Рассмотрим производную логарифма функции:
        \begin{equation*}
            \frac{\partial}{\partial x_1}\ln{f}=\frac{1}{f}\cdot\frac{\partial f}{\partial x_1}\Rightarrow\frac{\partial f}{\partial x_1}=f\frac{\partial}{\partial x_1}\ln{f}
        \end{equation*}
        Теперь используем это в нашем примере (сразу распишем логарифм произведений, как сумму логарифмов):
        \begin{equation*}
            \frac{\partial}{\partial x}\bigg(\ln{x}+\ln{y}+\ln{(x^2-y^2)}-\ln{(x^2+y^2)}\bigg)=\frac{1}{x}+\frac{2x}{x^2-y^2}-\frac{2x}{x^2+y^2}
        \end{equation*}
        }
        Теперь необходимо умножить на саму функцию и привести хоть к какому-нибудь виду:
        \begin{equation*}
            \frac{\partial f}{\partial x}=xy\frac{x^2-y^2}{x^2+y^2}\bigg(\frac{1}{x}+\frac{2x}{x^2-y^2}-\frac{2x}{x^2+y^2}\bigg)=\frac{y(x^2-y^2)}{x^2+y^2}+\frac{2x^2y}{x^2+y^2}-\frac{2x^2y(x^2-y^2)}{(x^2+y^2)^2}=
        \end{equation*}
        \begin{equation*}
            =\frac{y(x^4-y^4)+2x^2y(x^2+y^2)-2x^2y(x^2-y^2)}{(x^2+y^2)^2}=\frac{y(x^2+y^2)^2-2y^5}{(x^2+y^2)^2}=y-\frac{2y^5}{(x^2+y^2)^2}
        \end{equation*}
        Рассмотрим $\frac{\partial f}{\partial x}$ в точке $(0,y)$:
        \begin{equation*}
            \frac{\partial f(0,y)}{\partial x}=-y
        \end{equation*}
        Тогда получим, что
        \begin{equation*}
            \frac{\partial^2f(0,0)}{\partial y\partial x}=-1
        \end{equation*}
        Аналогичным способом рассмотрим $\frac{\partial f(0,0)}{\partial x\partial y}$ и получим, что
        \begin{equation*}
            \frac{\partial^2 f(0,0)}{\partial x\partial y}=1
        \end{equation*}
        Видим, что вообще переставлять производные нельзя. Однако мы знаем и примеры функций, в которых это делать можно.
        \par
        \noindent\hrulefill\noindent
        \par{\ }
        \par{\textbf{Теорема (перестановка производных)}\\
        Пусть $f$ определена на $V\subset\mathbb{R}^n$ и $V$ является открытым, существуют $f'_x$, $f'_y$, $f''_{xy}$, $f''_{yx}$ на $V$, причём $f''_{xy}$ и $f''_{yx}$ непрерывны в точке $(x_0,y_0)\in V$. Тогда $f''_{xy}(x_0,y_0)=f''_{yx}(x_0,y_0)$. \\
        \textbf{Доказательство: } рассмотрим функцию $W$
        \begin{equation}
            W=\bigg(f(x_0+\Delta x,y_0+\Delta y)-f(x_0+\Delta x,y_0)-f(x_0,y_0+\Delta y)+f(x_0,y_0)\bigg)\frac{1}{\Delta x\Delta y}
        \end{equation}
        \underline{\textit{1. Рассмотрим функцию $\phi$:}}
        \begin{equation}
            \phi(x)=\frac{f(x,y_0+\Delta y)-f(x,y_0)}{\Delta y}
        \end{equation}
        В силу существования первых проивзодных функции $f$ $\exists\phi'(x)$
        \begin{equation}
            \phi'(x)=\frac{f'_x(x,y_0+\Delta y)-f'_x(x,y_0)}{\Delta y}
        \end{equation}
        Подставим (2) в (1) и получим, что
        \begin{equation}
            W=\frac{\phi(x_0+\Delta x)-\phi(x_0)}{\Delta x}
        \end{equation}
        Так как $\phi$ непрерывна и дифференцируема во внутр. точках, то по теореме Лагранжа
        \begin{equation}
            W=\phi'(x_0+\theta_1\Delta x)
        \end{equation}
        Сейчас в (5) подставим формулу (3):
        \begin{equation}
            W=\frac{f'_x(x_0+\theta_1\Delta x,y_0+\Delta y)-f'_x(x_0+\theta_1\Delta x,y_0)}{\Delta y}=f''_{xy}(x_0+\theta_1\Delta x,y_0+\theta_2\Delta y)
        \end{equation}
        \underline{\textit{2. Рассмотрим функцию $\psi$:}}
        \begin{equation}
            \psi(y)=\frac{f(x_0+\Delta x, y)-f(x_0,y)}{\Delta x}
        \end{equation}
        В силу существования первых проивзодных функции $f$ $\exists\psi'(x)$
        \begin{equation}
            \psi'(x)=\frac{f'_y(x_0+\Delta x, y)-f'_y(x_0,y)}{\Delta x}
        \end{equation}
        Подставим (8) в (1) и получим, что
        \begin{equation}
            W=\frac{\psi(y_0+\Delta y)-\psi(y_0)}{\Delta y}
        \end{equation}
        Так как $\psi$ непрерывна и дифференцируема во внутр. точках, то по теореме Лагранжа
        \begin{equation}
            W=\psi'(y_0+\theta_3\Delta y)
        \end{equation}
        }
        Сейчас в (10) подставим формулу (8):
        \begin{equation}
            W=\frac{f'_y(x_0+\Delta x,y_0+\theta_3\Delta y)-f'_y(x_0,y_0+\theta_3\Delta y)}{\Delta x}=f''_{yx}(x_0+\theta_4\Delta x, y_0+\theta_3\Delta y)
        \end{equation}
        \underline{\textit{3. Перейдём к пределу:}}\\
        Пусть $\Delta x\to0$ и $\Delta y\to0$ - переход к пределу (он возможен, т.к. $f''_{xy}$ и $f''_{yx}$ непрерывны в точке $(x_0,y_0)$): из $W=(6)=(11)$ следует $f''_{xy}(x_0,y_0)=f''_{yx}(x_0,y_0)$.
        \par{{\raggedleft ч.т.д.\par}}
        \setcounter{equation}{0}
        \vspace{0.5cm}
        \par{\textbf{Обобщение: }Пусть $V\subset\mathbb{R}^n$. $C^k(V)$ - множество всех функций, имеющих все непрерывные частные производные до порядка $k$ включительно. По индукции из предыдущей теоремы можно показать, что порядок дифференцирования функции $f\in C^k(V)$ не важен.

        \subsection*{7. Мультииндексы}
        \addcontentsline{toc}{subsection}{7. Мультииндексы}
        \setcounter{equation}{0}
        \par{\ }
        \par{Введём следующие обозначения (мультииндексы): }
        \begin{itemize}
            \item \par{$\alpha=(\alpha_1,\ldots,\alpha_n)$, $\alpha_j=0,1,\ldots,n$}
            \item \par{$|\alpha|=\displaystyle\sum_{i=1}^n\alpha_i$}
            \item \par{$\displaystyle\partial^{\alpha}f(x)=\frac{\partial^{|\alpha|}f}{\partial x_1^{\alpha_1}\cdot\ldots\cdot\partial x_n^{\alpha_n}}$}
            \item \par{$\alpha+\beta=(\alpha_1+\beta_1,\ldots,\alpha_n+\beta_n)$}
            \item \par{$\alpha>\beta\Leftrightarrow\beta<\alpha\Leftrightarrow\alpha_i>\beta_i$ $\forall i=1,\ldots,n$}
            \item \par{$\alpha\geq\beta\Leftrightarrow\beta\leq\alpha\Leftrightarrow\alpha_i\geq\beta_i$ $\forall i=1,\ldots,n$}
            \item \par{$\alpha!=\alpha_1!\cdot\ldots\cdot\alpha_n!$}
            \item \par{$x^{\alpha}=x_1^{\alpha_1}\cdot\ldots\cdot x_n^{\alpha_n}$}
        \end{itemize}
        \vspace{0.5cm}
        \par{\textbf{Пример: }\\
        $\displaystyle(x+y)^{\alpha}=\sum_{\beta\leq\alpha}\frac{\alpha!}{\beta!(\alpha-\beta)!}x^{\beta}y^{\alpha-\beta}$\\
        \par{Что значит $\displaystyle\sum_{\beta\leq\alpha}$? Это означает, что мы рассматриваем все такие наборы $\beta=(\beta_1,\ldots,\beta_n)$, что $\beta_i\leq\alpha_i$, т.е. любую такую комбинацию. В итоге мы рассматриваем сумму по всем $\alpha_i$, где $\beta_i$ меняется от $0$ до $\alpha_i$.}
        \begin{equation*}
            (x+y)^{\alpha}=(x_1+y_1)^{\alpha_1}\cdot\ldots\cdot(x_n+y_n)^{\alpha_n}=\prod_{i=1}^n\bigg(\sum_{\beta_i=0}^{\alpha_i}\frac{\alpha_i!}{\beta_i!(\alpha_i-\beta_i)!}x_i^{\beta_i}y_i^{\alpha_i-\beta_i}\bigg)
        \end{equation*}
        Пусть $n=1$ - очевидно, что $\alpha=(\alpha_1)$, тогда
        \begin{equation*}
            (x+y)^{\alpha}=(x_1+y_1)^{\alpha_1}=\sum_{\beta_1=0}^{\alpha_1}\frac{\alpha_1!}{\beta_1!(\alpha_1-\beta_1)!}x_1^{\beta_1}y_1^{\alpha_1-\beta_1}
        \end{equation*}
        Это мы знаем из обычного бинома Ньютона. Ну очевидно, что это сводится к нашим мультииндексам, ведь суммируем по всем $\beta\leq\alpha$, где $\beta=(\beta_1)$, $\alpha=(\alpha_1)$ и $\beta_1$ пробегает от 0 до $\alpha_1$, т.е. как раз по $\beta_1\leq\alpha_1$.
        Пусть $n=2$, тогда $\alpha=(\alpha_1,\alpha_2)$ и $\beta=(\beta_1,\beta_2)$, тогда
        \begin{equation*}
            (x+y)^{\alpha}=(x_1+y_1)^{\alpha_1}\cdot(x_2+y_2)^{\alpha_2}=\sum_{\beta_1=0}^{\alpha_1}\frac{\alpha_1!}{\beta_1!(\alpha_1-\beta_1)!}x_1^{\beta_1}y_1^{\alpha_1-\beta_1}\cdot\sum_{\beta_2=0}^{\alpha_2}\frac{\alpha_2!}{\beta_2!(\alpha_2-\beta_2)!}x_2^{\beta_2}y_2^{\alpha_2-\beta_2}=
        \end{equation*}
        \begin{equation*}
            =\sum_{\beta_1=0}^{\alpha_1}\sum_{\beta_2=0}^{\alpha_2}\frac{\alpha_1!}{\beta_1!(\alpha_1-\beta_1)!}x_1^{\beta_1}y_1^{\alpha_1-\beta_1}\cdot\frac{\alpha_2!}{\beta_2!(\alpha_2-\beta_2)!}x_2^{\beta_2}y_2^{\alpha_2-\beta_2}=
        \end{equation*}
        \begin{equation*}
            =\sum_{\beta_1=0}^{\alpha_1}\sum_{\beta_2=0}^{\alpha_2}\frac{\alpha_1!\alpha_2!}{\beta_1!\beta_2!(\alpha_1-\beta_1)!(\alpha_2-\beta_2)!}x_1^{\beta_1}x_2^{\beta_2}y_1^{\alpha_1-\beta_1}y_2^{\alpha_2-\beta_2}
        \end{equation*}
        А сейчас внимательно посмотрим на определения мультииндексов: факториалы сворачиваются в мультииндекс, степени сворачиваются аналогично. Альфа и бета в степенях $x$ и $y$ заносятся в сумму по определению суммы, а суммирование превращается в $\beta\leq\alpha$. Ведь что такое суммирование по $\beta_1$, а затем по $\beta_2$? Это комбинации $(0,0)$, $(0,1)$, $\ldots$, $(0,\alpha_2)$, $(1,0)$, $(1,1)$, $\ldots$, $(1,\alpha_2)$, $\ldots$, $(\alpha_1,0)$, $(\alpha_1,1)$, $\ldots$, $(\alpha_1,\alpha_2)$ - а это и есть определение всевозможных комбинаций, что $\beta\leq\alpha$. Ну и всё, отсюда получаем формулу для $n=2$:
        \begin{equation*}
             \displaystyle(x+y)^{\alpha}=\sum_{\beta\leq\alpha}\frac{\alpha!}{\beta!(\alpha-\beta)!}x^{\beta}y^{\alpha-\beta}
        \end{equation*}
        }

        \subsection*{8. Формула Тейлора}
        \addcontentsline{toc}{subsection}{8. Формула Тейлора}
        \setcounter{equation}{0}
        \par{\ }
        \par{Пусть $f$ определена в шаре $B_{\delta}(x_0)\subset\mathbb{R}^n$ и в нём $N+1$ раз непрерывно дифференцируема ($f\in C^{N+1}[B_{\delta}(x_0)]$), тогда $\forall x\in B_{\delta}(x_0)$ справедлива формула Тейлора:
        \begin{equation}
            f(x)=\sum_{|\alpha|\leq N}\frac{1}{\alpha!}\frac{\partial^{|\alpha|}f(x_0)}{\partial x_1^{\alpha_1}\cdot\ldots\cdot\partial x_n^{\alpha_n}}(x-x_0)^{\alpha}+R_{N+1}(x)
        \end{equation}
        где $R_{N+1}(x)$ - остаточный член равен
        \begin{equation}
            R_{N+1}(x)=\sum_{|\alpha|=N+1}\frac{N+1}{\alpha!}\cdot\int_0^1(1-s)^N\frac{\partial^{|\alpha|}f(x_0+s(x-x_0))}{\partial x_1^{\alpha_1}\cdot\ldots\cdot\partial x_n^{\alpha_n}}\cdot(x-x_0)^{\alpha}ds
        \end{equation}
        }
        \par
        \noindent\hrulefill\noindent
        \par{\ }
        \par{Разберёмся с формулой Тейлора и мультииндексами: давайте запишем формулу тейлора в привычной формулой: в привычной формуле $i$-ой производной соответствует деление на $i!$ и возведение в степень $i$.
        \newpage
        Обозначим за $p=x-x_0$ и ведём производные следующим образом:
        \begin{equation*}
            \phi'(t)=\sum_{i=1}^n\frac{\partial f}{\partial x_i}(x_0+tp)\cdot p_i
        \end{equation*}
        \begin{equation*}
            \phi^{(2)}(t)=\sum_{i=1}^n\sum_{j=1}^n\frac{\partial^2 f}{\partial x_i\partial x_j}(x_0+tp)\cdot p_ip_j
        \end{equation*}
        Тогда для производной порядка $s$ верна следующая формула:
        \begin{equation}
            \phi^{(m)}(t)=\underbrace{\sum_{i=1}^n\ldots\sum_{k=1}^n}_{m}\frac{\partial^m f}{\underbrace{\partial x_i\ldots\partial x_k}_{m}}(x_0+tp)\cdot \underbrace{p_i\ldots p_k}_{m}
        \end{equation}
        Т.о. мы расписали $p$ и $x$ по координатам $p_i$ и $x_i$ в $\mathbb{R}^n$. В итоге наша формула Тейлора превращается в следующее, где соответствующая производная определяется формулой (3):
        \begin{equation}
            f(x)=f(x_0)+\frac{\phi'(0)}{1!}+\frac{\phi^{(2)}(0)}{2!}+\ldots+\frac{\phi^{(N)}(0)}{N!}+\frac{1}{N!}\int_{0}^{1}(1-s)^N\phi^{N+1}(s)ds
        \end{equation}
        Теперь рассмотрим нашу формулу: в нашей формуле $|\alpha|$ играет роль $m$ в формуле, а суммирование по $|\alpha|\leq N$ - это аналог суммы производных (4) и (3). Объяснить, почему факториалы у наших производных совпадают будет сложнее, но общая идея заключается в том, что в формуле (3) можно переставлять производные местами, а из-за этого фактически наша производная встречается много раз. \\
        \\
        Рассмотрим пример для функции двух переменных в разложении до третьей производной (остаток рассматривать не будем - это упражнение):\\
        $\displaystyle \phi'(0)=\frac{\partial f(x_0)}{\partial x_1}\cdot p_1+\frac{\partial f(x_0)}{\partial x_2}\cdot p_2$ \\
        \\ \\
        $\displaystyle \phi^{(2)}(0)=\frac{\partial^2f(x_0)}{\partial x_{1}\partial x_{1}}\cdot p_1p_1+\frac{\partial^2f(x_0)}{\partial x_{1}\partial x_{2}}\cdot p_1p_2+\frac{\partial^2f(x_0)}{\partial x_{2}\partial x_{1}}\cdot p_2p_1+\frac{\partial^2f(x_0)}{\partial x_{2}\partial x_{2}}\cdot p_2p_2=$\\ \\
        $\displaystyle=\frac{\partial^2f(x_0)}{\partial x_{1}^2}\cdot p_1^2+2\frac{\partial^2f(x_0)}{\partial x_{1}\partial x_{2}}\cdot p_1p_2+\frac{\partial^2f(x_0)}{\partial x_{2}^2}\cdot p_2^2$ \\ \\ \\
        - здесь мы воспользовались тем, что производные можно переставлять, а ещё сгруппировали члены. Аналогично получим формулу для третей производной (но расспишем её менее подробно):\\
        $\displaystyle \phi^{(3)}(0)=\frac{\partial^3f(x_0)}{\partial x_1^3}\cdot p_1^3+3\frac{\partial^3f(x_0)}{\partial x_1^2\partial x_2}\cdot p_1^2p_2+3\frac{\partial^3f(x_0)}{\partial x_1\partial x_2^2}\cdot p_1p_2^2+\frac{\partial^3f(x_0)}{\partial x_2^3}\cdot p_2^3$
        }
